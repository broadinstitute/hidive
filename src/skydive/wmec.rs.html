1//! This module implements the `WhatsHap` phasing algorithm as described in:
2//! Murray Patterson, Tobias Marschall, Nadia Pisanti, Leo van Iersel, Leen Stougie, et al.
3//! `WhatsHap`: Weighted Haplotype Assembly for Future-Generation Sequencing Reads.
4//! Journal of Computational Biology, 2015, 22 (6), pp.498-509.
5//! DOI: 10.1089/cmb.2014.0157
6//! HAL ID: hal-01225988
7

17#[derive(Debug)]
18pub struct WMECData {
19    pub reads: Vec<Vec<Option<u8>>>,        // Reads matrix where None represents missing data
20    pub confidences: Vec<Vec<Option<u32>>>, // Confidence degrees matrix
21    pub num_snps: usize,                    // Number of SNP positions
22}
23
24impl WMECData {
25    // Initialize the data structure with given reads and confidences
26    #[must_use]
27    pub fn new(reads: Vec<Vec<Option<u8>>>, confidences: Vec<Vec<Option<u32>>>) -> Self {
28        let num_snps = reads[0].len();
29        WMECData { reads, confidences, num_snps }
30    }
31    
32    // Function to compute W^0(j, R) and W^1(j, R)
33    // Cost to set all fragments in set R to 0 or 1 at SNP j
34    #[must_use]
35    pub fn compute_costs(&self, snp: usize, set_r: &BTreeSet<usize>) -> (u32, u32) {
36        let mut w0 = 0; // Cost for setting to 0
37        let mut w1 = 0; // Cost for setting to 1
38
39        for &read_index in set_r {
40            if let Some(allele) = self.reads[read_index][snp] {
41                if let Some(confidence) = self.confidences[read_index][snp] {
42                    if allele == 0 {
43                        w1 += confidence; // Cost of flipping 0 -> 1
44                    } else {
45                        w0 += confidence; // Cost of flipping 1 -> 0
46                    }
47                }
48            }
49        }
50
51        (w0, w1)
52    }
53    
54    // Calculate minimum correction cost Delta C(j, (R, S))
55    #[must_use]
56    pub fn delta_c(&self, snp: usize, r: &BTreeSet<usize>, s: &BTreeSet<usize>) -> u32 {
57        let (w0_r, w1_r) = self.compute_costs(snp, r);
58        let (w0_s, w1_s) = self.compute_costs(snp, s);
59
60        // Given a bipartition (R, S) of F(j), the minimum cost to make position j conflict-free is:
61        std::cmp::min(w0_r, w1_r) + std::cmp::min(w0_s, w1_s)
62
63        // Alternatively, under the all heterozygous assumption, where one wants to enforce all SNPs
64        // to be heterozygous, the equation becomes:
65        // std::cmp::min(w0_r + w1_r, w0_s + w1_s)
66    }
67
68    // Write reads matrix to a file in tab-separated format
69    pub fn write_reads_matrix(&self, path: &str) -> std::io::Result<()> {
70        let mut file = File::create(path)?;
71
72        // Write header row with SNP positions
73        for j in 0..self.num_snps {
74            write!(file, "\tSNP_{}", j)?;
75        }
76        writeln!(file)?;
77
78        // Write each read's data
79        for (i, (read, conf)) in self.reads.iter().zip(self.confidences.iter()).enumerate() {
80            write!(file, "Read_{}", i)?;
81            for (allele, qual) in read.iter().zip(conf.iter()) {
82                if let Some(allele) = allele {
83                    if *allele == 0 { write!(file, "\t0,{}", qual.unwrap())? };
84                    if *allele == 1 { write!(file, "\t1,{}", qual.unwrap())? };
85                } else {
86                    write!(file, "\t-,1")?;
87                }
88            }
89            writeln!(file)?;
90        }
91
92        Ok(())
93    }
94}

96// Function to generate all bipartitions of a set
97fn generate_bipartitions(set: &BTreeSet<usize>) -> Vec<(BTreeSet<usize>, BTreeSet<usize>)> {
98    let mut partitions = vec![];
99    let set_vec: Vec<_> = set.iter().collect();
100
101    let num_partitions = 1 << set_vec.len(); // 2^|set|
102    for i in 0..num_partitions {
103        let mut r = BTreeSet::new();
104        let mut s = BTreeSet::new();
105
106        for (j, &elem) in set_vec.iter().enumerate() {
107            if i & (1 << j) == 0 {
108                r.insert(*elem);
109            } else {
110                s.insert(*elem);
111            }
112        }
113
114        partitions.push((r, s));
115    }
116
117    partitions
118}

120// Function to initialize the DP table for SNP 0
121fn initialize_dp(data: &WMECData) -> (HashMap<(usize, BTreeSet<usize>, BTreeSet<usize>), u32>, HashMap<(usize, BTreeSet<usize>, BTreeSet<usize>), Option<(BTreeSet<usize>, BTreeSet<usize>)>>) {
122    let mut dp: HashMap<(usize, BTreeSet<usize>, BTreeSet<usize>), u32> = HashMap::new();
123    let mut backtrack: HashMap<(usize, BTreeSet<usize>, BTreeSet<usize>), Option<(BTreeSet<usize>, BTreeSet<usize>)>> = HashMap::new();
124
125    let active_fragments: BTreeSet<usize> = data.reads.iter().enumerate()
126        .filter(|(_, read)| read[0].is_some()) // Only consider fragments covering SNP 0
127        .map(|(index, _)| index)
128        .collect();
129
130    let partitions = generate_bipartitions(&active_fragments);
131    for (r, s) in partitions {
132        let cost = data.delta_c(0, &r, &s);
133        dp.insert((0, r.clone(), s.clone()), cost);
134        backtrack.insert((0, r.clone(), s.clone()), None);
135    }
136
137    (dp, backtrack)
138}

140// Function to update the DP table for each SNP position
141fn update_dp_old(data: &WMECData, dp: &mut HashMap<(usize, BTreeSet<usize>, BTreeSet<usize>), u32>, backtrack: &mut HashMap<(usize, BTreeSet<usize>, BTreeSet<usize>), Option<(BTreeSet<usize>, BTreeSet<usize>)>>, snp: usize) {
142    let active_fragments: BTreeSet<usize> = data.reads.iter().enumerate()
143        .filter(|(_, read)| read[snp].is_some()) // Only consider fragments covering SNP
144        .map(|(index, _)| index)
145        .collect();
146    let partitions = generate_bipartitions(&active_fragments);
147
148    for (r, s) in &partitions {
149        let delta_cost = data.delta_c(snp, r, s);
150        let mut min_cost = u32::MAX;
151        let mut best_bipartition = None;
152
153        let prev_active_fragments: BTreeSet<usize> = data.reads.iter().enumerate()
154            .filter(|(_, read)| read[snp - 1].is_some()) // Fragments covering the previous SNP
155            .map(|(index, _)| index)
156            .collect();
157        
158        for (prev_r, prev_s) in generate_bipartitions(&prev_active_fragments) {
159            let r_compatible = r.intersection(&prev_active_fragments).all(|&x| prev_r.contains(&x));
160            let s_compatible = s.intersection(&prev_active_fragments).all(|&x| prev_s.contains(&x));
161
162            if r_compatible && s_compatible {
163                if let Some(&prev_cost) = dp.get(&(snp - 1, prev_r.clone(), prev_s.clone())) {
164                    let current_cost = delta_cost + prev_cost;
165
166                    if current_cost < min_cost {
167                        min_cost = current_cost;
168                        best_bipartition = Some((prev_r.clone(), prev_s.clone()));
169                    }
170                }
171            }
172        }
173
174        dp.insert((snp, r.clone(), s.clone()), min_cost);
175        backtrack.insert((snp, r.clone(), s.clone()), best_bipartition);
176    }
177}

179fn update_dp(data: &WMECData, dp: &mut HashMap<(usize, BTreeSet<usize>, BTreeSet<usize>), u32>, backtrack: &mut HashMap<(usize, BTreeSet<usize>, BTreeSet<usize>), Option<(BTreeSet<usize>, BTreeSet<usize>)>>, snp: usize) {
180    let active_fragments: BTreeSet<usize> = data.reads.iter().enumerate()
181        .filter(|(_, read)| read[snp].is_some())
182        .map(|(index, _)| index)
183        .collect();
184    let partitions = generate_bipartitions(&active_fragments);
185    
186    // Pre-compute prev_active_fragments since it's used by all iterations
187    let prev_active_fragments: BTreeSet<usize> = data.reads.iter().enumerate()
188        .filter(|(_, read)| read[snp - 1].is_some())
189        .map(|(index, _)| index)
190        .collect();
191
192    // Collect results in parallel
193    let results: Vec<_> = partitions.par_iter()
194        .map(|(r, s)| {
195            let delta_cost = data.delta_c(snp, r, s);
196            let mut min_cost = u32::MAX;
197            let mut best_bipartition = None;
198            
199            for (prev_r, prev_s) in generate_bipartitions(&prev_active_fragments) {
200                let r_compatible = r.intersection(&prev_active_fragments).all(|&x| prev_r.contains(&x));
201                let s_compatible = s.intersection(&prev_active_fragments).all(|&x| prev_s.contains(&x));
202
203                if r_compatible && s_compatible {
204                    if let Some(&prev_cost) = dp.get(&(snp - 1, prev_r.clone(), prev_s.clone())) {
205                        let current_cost = delta_cost + prev_cost;
206
207                        if current_cost < min_cost {
208                            min_cost = current_cost;
209                            best_bipartition = Some((prev_r.clone(), prev_s.clone()));
210                        }
211                    }
212                }
213            }
214
215            ((r.clone(), s.clone()), (min_cost, best_bipartition))
216        })
217        .collect();
218
219    // Update the hashmaps with the results
220    for ((r, s), (min_cost, best_bipartition)) in results {
221        dp.insert((snp, r.clone(), s.clone()), min_cost);
222        backtrack.insert((snp, r, s), best_bipartition);
223    }
224}

226fn backtrack_haplotypes(data: &WMECData, dp: &HashMap<(usize, BTreeSet<usize>, BTreeSet<usize>), u32>, backtrack: &HashMap<(usize, BTreeSet<usize>, BTreeSet<usize>), Option<(BTreeSet<usize>, BTreeSet<usize>)>>) -> (Vec<u8>, Vec<u8>, BTreeSet<usize>, BTreeSet<usize>) {
227    let mut best_cost = u32::MAX;
228    let mut best_bipartition = None;
229
230    // Restrict processing to reads that span variants within the window.
231    let final_active_fragments: BTreeSet<usize> = data.reads.iter().enumerate()
232        .filter(|(_, read)| read[data.num_snps - 1].is_some())
233        .map(|(index, _)| index)
234        .collect();
235
236    for (r, s) in generate_bipartitions(&final_active_fragments) {
237        if let Some(&cost) = dp.get(&(data.num_snps - 1, r.clone(), s.clone())) {
238            if cost < best_cost {
239                best_cost = cost;
240                best_bipartition = Some((r, s));
241            }
242        }
243    }
244
245    let mut haplotype1 = vec![0; data.num_snps];
246    let mut haplotype2 = vec![0; data.num_snps];
247    let mut current_bipartition = best_bipartition.unwrap();
248
249    for snp in (0..data.num_snps).rev() {
250        let (r, s) = &current_bipartition;
251        let (w0_r, w1_r) = data.compute_costs(snp, r);
252        let (w0_s, w1_s) = data.compute_costs(snp, s);
253
254        if w0_r + w1_s <= w1_r + w0_s {
255            haplotype1[snp] = 0;
256            haplotype2[snp] = 1;
257        } else {
258            haplotype1[snp] = 1;
259            haplotype2[snp] = 0;
260        }
261
262        if snp > 0 {
263            if let Some(prev_bipartition) = backtrack.get(&(snp, r.clone(), s.clone())).and_then(|x| x.as_ref()) {
264                current_bipartition = prev_bipartition.clone();
265            } else {
266                // This should not happen if the DP table is correctly filled
267                panic!("No valid previous bipartition found for SNP {snp}");
268            }
269        }
270    }
271
272    crate::elog!("wmec score: {}", best_cost);
273
274    (haplotype1, haplotype2, current_bipartition.0, current_bipartition.1)
275}

277// Main function to perform WMEC using dynamic programming
278#[must_use]
279pub fn phase(data: &WMECData) -> (Vec<u8>, Vec<u8>, BTreeSet<usize>, BTreeSet<usize>) {
280    let (mut dp, mut backtrack) = initialize_dp(data);
281
282    for snp in 1..data.num_snps {
283        update_dp(data, &mut dp, &mut backtrack, snp);
284    }
285
286    backtrack_haplotypes(data, &dp, &backtrack)
287}

289#[must_use]
290pub fn phase_all(data: &WMECData, window: usize, stride: usize) -> (Vec<u8>, Vec<u8>, BTreeSet<usize>, BTreeSet<usize>) {
291    data.write_reads_matrix("mat.tsv");
292
293    // First, collect all window ranges
294    let mut windows: Vec<_> = (0..data.num_snps)
295        .step_by(stride)
296        .map(|start| (start, (start + window).min(data.num_snps)))
297        .collect();
298
299    // Filter out last window if it completely overlaps with previous window
300    if windows.len() > 1 {
301        if let Some(&(_, prev_end)) = windows.get(windows.len() - 2) {
302            if let Some(&(_, last_end)) = windows.last() {
303                if last_end == prev_end {
304                    windows.pop();
305                }
306            }
307        }
308    }
309
310    let pb = crate::utils::default_bounded_progress_bar("Processing windows", windows.len() as u64);
311
312    // Process windows in parallel
313    let haplotype_pairs: Vec<_> = windows
314        // .iter()
315        .par_iter()
316        .progress_with(pb)
317        .map(|&(start, end)| {
318            let (window_indices, window_reads): (Vec<usize>, Vec<Vec<Option<u8>>>) = data.reads.iter().zip(data.confidences.iter()).enumerate()
319                .map(|(read_idx, (read, confidence))| {
320                    let window_read = read[start..end].to_vec();
321                    let window_confidence = confidence[start..end].to_vec();
322
323                    let none_count = window_read.iter().filter(|x| x.is_none()).count();
324                    let lowqual_count = window_confidence.iter().filter(|&&x| x.is_some() && x.unwrap() == 0).count();
325
326                    (none_count + lowqual_count, read_idx, window_read, window_confidence)
327                })
328                .collect::<Vec<_>>()
329                .into_iter()
330                .sorted_by_key(|(bad_count, _, _, _)| *bad_count)
331                .take(12)
332                .map(|(_, read_idx, window_read, _)| (read_idx, window_read))
333                .unzip();
334
335            let (_, window_confidences): (Vec<usize>, Vec<Vec<Option<u32>>>) = data.reads.iter().zip(data.confidences.iter()).enumerate()
336                .map(|(read_idx, (read, confidence))| {
337                    let window_read = read[start..end].to_vec();
338                    let window_confidence = confidence[start..end].to_vec();
339
340                    let none_count = window_read.iter().filter(|x| x.is_none()).count();
341                    let lowqual_count = window_confidence.iter().filter(|&&x| x.is_some() && x.unwrap() == 0).count();
342
343                    (none_count + lowqual_count, read_idx, window_read, window_confidence)
344                })
345                .collect::<Vec<_>>()
346                .into_iter()
347                .sorted_by_key(|(bad_count, _, _, _)| *bad_count)
348                .take(12)
349                .map(|(_, read_idx, _, window_confidences)| (read_idx, window_confidences))
350                .unzip();
351
352            crate::elog!("window_reads: {} {} {}", start, end, window_reads.len());
353            crate::elog!("window_indices: {} {} {:?}", start, end, window_indices);
354
355            let window_data = WMECData::new(window_reads, window_confidences);
356            
357            let (mut dp, mut backtrack) = initialize_dp(&window_data);
358            for snp in 1..window_data.num_snps {
359                update_dp(&window_data, &mut dp, &mut backtrack, snp);
360            }
361
362            let (hap1, hap2, part1, part2) = backtrack_haplotypes(&window_data, &dp, &backtrack);
363
364            let re1: BTreeSet<_> = part1.iter().map(|&i| window_indices[i as usize]).collect();
365            let re2: BTreeSet<_> = part2.iter().map(|&i| window_indices[i as usize]).collect();
366
367            crate::elog!("window_re1: {} {} {:?} {:?}", start, end, part1, re1);
368            crate::elog!("window_re2: {} {} {:?} {:?}", start, end, part2, re2);
369
370            (hap1, hap2, re1, re2)
371        })
372        .collect();
373
374    let mut haplotype1 = Vec::new();
375    let mut haplotype2 = Vec::new();
376    let mut part1 = BTreeSet::new();
377    let mut part2 = BTreeSet::new();
378
379    let overlap = window - stride;
380
381    let mut all_read_assignments = BTreeMap::new();
382
383    let mut i = 0;
384    for (hap1, hap2, reads1, reads2) in haplotype_pairs {
385        // crate::elog!("{}", i);
386        // crate::elog!("{:?}", reads1);
387        // crate::elog!("{:?}", reads2);
388
389        i += 1;
390
391        if haplotype1.len() == 0 {
392            for read_num in reads1.clone() {
393                all_read_assignments.entry(read_num).or_insert(vec![]).push(1);
394            }
395
396            for read_num in reads2.clone() {
397                all_read_assignments.entry(read_num).or_insert(vec![]).push(2);
398            }
399
400            haplotype1 = hap1;
401            haplotype2 = hap2;
402            part1 = reads1;
403            part2 = reads2;
404        } else {
405            // Compare overlap regions to determine orientation
406            crate::elog!("haplotypes: {} {} {}", haplotype1.len(), haplotype2.len(), overlap);
407
408            let new_overlap_len = std::cmp::min(overlap, hap1.len());
409            let h1_overlap = &haplotype1[haplotype1.len()-new_overlap_len..];
410            let h2_overlap = &haplotype2[haplotype2.len()-new_overlap_len..];
411            let new_overlap = &hap1[..new_overlap_len];
412
413            // Count matches between overlapping regions
414            let h1_matches = h1_overlap.iter().zip(new_overlap.iter())
415                .filter(|(a,b)| a == b)
416                .count();
417            let h2_matches = h2_overlap.iter().zip(new_overlap.iter())
418                .filter(|(a,b)| a == b)
419                .count();
420
421            // Append new haplotypes in correct orientation based on best overlap match
422            if h1_matches >= h2_matches {
423                haplotype1.extend_from_slice(&hap1[std::cmp::min(overlap, hap1.len())..]);
424                haplotype2.extend_from_slice(&hap2[std::cmp::min(overlap, hap2.len())..]);
425
426                for read_num in reads1.clone() {
427                    all_read_assignments.entry(read_num).or_insert(vec![]).push(1);
428                }
429
430                for read_num in reads2.clone() {
431                    all_read_assignments.entry(read_num).or_insert(vec![]).push(2);
432                }
433            } else {
434                haplotype1.extend_from_slice(&hap2[std::cmp::min(overlap, hap2.len())..]);
435                haplotype2.extend_from_slice(&hap1[std::cmp::min(overlap, hap1.len())..]);
436
437                for read_num in reads1.clone() {
438                    all_read_assignments.entry(read_num).or_insert(vec![]).push(2);
439                }
440
441                for read_num in reads2.clone() {
442                    all_read_assignments.entry(read_num).or_insert(vec![]).push(1);
443                }
444            }
445        }
446    }
447
448    let mut final_part1 = BTreeSet::new();
449    let mut final_part2 = BTreeSet::new();
450
451    crate::elog!("all_read_assignments: {:?}", &all_read_assignments);
452
453    for (read_num, assignments) in all_read_assignments {
454        // Count frequency of assignments (1 or 2) for this read
455        let most_common = assignments.iter()
456            .fold(HashMap::new(), |mut counts, &value| {
457                *counts.entry(value).or_insert(0) += 1;
458                counts
459            })
460            .into_iter()
461            .max_by_key(|&(_, count)| count)
462            .map(|(value, _)| value)
463            .unwrap_or(0); // Default to 0 if no assignments
464
465        if most_common == 1 {
466            final_part1.insert(read_num);
467        } else if most_common == 2 {
468            final_part2.insert(read_num);
469        }
470    }
471
472    (haplotype1, haplotype2, final_part1, final_part2)
473}
