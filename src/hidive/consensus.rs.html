18pub fn start(
19    output: &PathBuf,
20    loci_list: &Vec<String>,
21    reference_fasta_path: &PathBuf,
22    short_read_fasta_path: &PathBuf,
23    bam_path: &PathBuf,
24) {
25    // Get the system's temporary directory path
26    let cache_path = std::env::temp_dir();
27    skydive::elog!("Intermediate data will be stored at {:?}.", cache_path);
28
29    // Parse reference sequence file path.
30    // let reference_seq_urls = skydive::parse::parse_file_names(&[reference_fasta_path.clone()]);
31    // let reference_seq_url = reference_seq_urls.iter().next().unwrap();
32    // let fasta = skydive::stage::open_fasta(&reference_seq_url).unwrap();
33
34    skydive::elog!("Processing short-read sample {}...", short_read_fasta_path.display());
35    let all_sr_seqs = skydive::utils::read_fasta(&vec![short_read_fasta_path.clone()]);
36    skydive::elog!(" - {} short reads loaded", all_sr_seqs.len());
37
38    // Parse BAM file path.
39    let bam_urls = skydive::parse::parse_file_names(&[bam_path.clone()]);
40    let bam_url = bam_urls.iter().next().unwrap();
41
42    // Iterate over loc
43    let loci = skydive::parse::parse_loci(loci_list, 0).into_iter().collect::<Vec<_>>();
44
45    // Open FASTA file for writing.
46    let mut buf_writer = BufWriter::new(File::create(output).unwrap());
47    let mut fasta_writer = fasta::Writer::new(&mut buf_writer);
48
49    for (chr, start_pos, stop_pos, name) in loci {
50        skydive::elog!("Processing locus {} ({}:{}-{})...", name, chr, start_pos, stop_pos);
51
52        let mut sg = spoa::Graph::new();
53
54        // The BAM reader gets renewed for each locus, but it's fast to open.
55        let mut bam = skydive::stage::open_bam(&bam_url).unwrap();
56        let _ = bam.fetch(FetchDefinition::RegionString(chr.as_bytes(), start_pos as i64, stop_pos as i64));
57
58        for read in bam.records().filter(|r| r.is_ok()).flatten() {
59            let mut la = spoa::AlignmentEngine::new(AlignmentType::kOV, 5, -4, -8, -6, -8, -4);
60
61            let subseq = read.seq().as_bytes();
62
63            let seq_cstr = std::ffi::CString::new(subseq.clone()).unwrap();
64            let seq_qual = std::ffi::CString::new(vec![b'I'; subseq.len()]).unwrap();
65            let a = la.align(seq_cstr.as_ref(), &sg);
66            sg.add_alignment(&a, seq_cstr.as_ref(), seq_qual.as_ref());
67        }
68
69        let consensus_cstr = sg.consensus();
70
71        let record = fasta::Record::with_attrs(name.as_str(), None, consensus_cstr.to_str().unwrap().as_bytes());
72        fasta_writer.write_record(&record).unwrap();
73    }
74
75    // let _ = fasta_writer.flush();
