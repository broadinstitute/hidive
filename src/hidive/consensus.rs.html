21pub fn start(
22    output: &PathBuf,
23    loci_list: &Vec<String>,
24    // reference_fasta_path: &PathBuf,
25    // short_read_fasta_path: &PathBuf,
26    bam_path: &PathBuf,
27) {
28    // Get the system's temporary directory path
29    let cache_path = std::env::temp_dir();
30    skydive::elog!("Intermediate data will be stored at {:?}.", cache_path);
31
32    // Parse reference sequence file path.
33    // let reference_seq_urls = skydive::parse::parse_file_names(&[reference_fasta_path.clone()]);
34    // let reference_seq_url = reference_seq_urls.iter().next().unwrap();
35    // let fasta = skydive::stage::open_fasta(&reference_seq_url).unwrap();

37    // skydive::elog!("Processing short-read sample {}...", short_read_fasta_path.display());
38    // let all_sr_seqs = skydive::utils::read_fasta(&vec![short_read_fasta_path.clone()]);
39    // skydive::elog!(" - {} short reads loaded", all_sr_seqs.len());
40
41    // Parse BAM file path.
42    let bam_urls = skydive::parse::parse_file_names(&[bam_path.clone()]);
43    let bam_url = bam_urls.iter().next().unwrap();
44
45    // Iterate over loc
46    let loci = skydive::parse::parse_loci(loci_list, 0)
47        .into_iter()
48        .collect::<Vec<_>>();
49
50    // Open FASTA file for writing.
51    let mut buf_writer = BufWriter::new(File::create(output).unwrap());
52    let mut fasta_writer = fasta::Writer::new(&mut buf_writer);
53
54    for (chr, start_pos, stop_pos, name) in loci {
55        skydive::elog!(
56            "Processing locus {} ({}:{}-{})...",
57            name,
58            chr,
59            start_pos,
60            stop_pos
61        );
62
63        let mut sg = spoa::Graph::new();
64
65        // The BAM reader gets renewed for each locus, but it's fast to open.
66        let mut bam = skydive::stage::open_bam(&bam_url).unwrap();
67        let _ = bam.fetch(FetchDefinition::RegionString(
68            chr.as_bytes(),
69            start_pos as i64,
70            stop_pos as i64,
71        ));
72
73        for read in bam.records().filter(|r| r.is_ok()).flatten() {
74            let mut la = spoa::AlignmentEngine::new(AlignmentType::kOV, 5, -4, -8, -6, -8, -4);
75
76            let subseq = read.seq().as_bytes();
77
78            let seq_cstr = std::ffi::CString::new(subseq.clone()).unwrap();
79            let seq_qual = std::ffi::CString::new(vec![b'I'; subseq.len()]).unwrap();
80            let a = la.align(seq_cstr.as_ref(), &sg);
81            sg.add_alignment(&a, seq_cstr.as_ref(), seq_qual.as_ref());
82        }
83
84        let consensus_cstr = sg.consensus();
85
86        let record = fasta::Record::with_attrs(
87            name.as_str(),
88            None,
89            consensus_cstr.to_str().unwrap().as_bytes(),
90        );
91        fasta_writer.write_record(&record).unwrap();
92    }
93
