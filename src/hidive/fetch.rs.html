9/// Starts the data fetching process.
10///
11/// # Arguments
12///
13/// * `output` - A reference to a PathBuf that specifies the output file path.
14/// * `loci_list` - A reference to a vector of strings, each representing a genomic locus.
15/// * `seq_paths` - A reference to a vector of PathBufs, each representing a path to a BAM or FASTA file.
16///
17/// # Panics
18///
19/// Panics if any locus in `loci_list` cannot be parsed.
20pub fn start(
21    output: &PathBuf,
22    loci_list: &Vec<String>,
23    padding: u64,
24    haplotype: Option<u8>,
25    seq_paths: &Vec<PathBuf>,
26) {
27    let loci = skydive::parse::parse_loci(loci_list, padding);
28    let seq_urls = skydive::parse::parse_file_names(seq_paths);
29
30    // Get the system's temporary directory path
31    let cache_path = std::env::temp_dir();
32    skydive::elog!("Intermediate data will be stored at {:?}.", cache_path);
33
34    // Call the stage_data function from the skydive module to process and stage the data
35    skydive::elog!("Fetching data...");
36    let r = skydive::stage::stage_data(output, &loci, &seq_urls, false, haplotype, &cache_path);
37
38    match r {
39        Ok(n) => {
40            skydive::elog!(
41                "Wrote {} sequences to '{}'.",
42                n,
43                output.absolutize().unwrap().to_str().unwrap()
44            )
45        }
46        Err(_) => {
47            panic!("Failed to write output FASTA.")
48        }
49    }
50}
