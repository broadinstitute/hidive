39pub fn start(
40    output: &PathBuf,
41    kmer_size: usize,
42    min_kmers_pct: usize,
43    search_option: SearchOption,
44    ref_path: Option<PathBuf>,
45    loci: Option<Vec<String>>,
46    fasta_paths: &Vec<PathBuf>,
47    seq_paths: &Vec<PathBuf>,
48) {
49    let fasta_urls = skydive::parse::parse_file_names(fasta_paths);
50    let seq_urls = skydive::parse::parse_file_names(seq_paths);
51
52    // Get the system's temporary directory path
53    let cache_path = std::env::temp_dir();
54    skydive::elog!("Intermediate data will be stored at {:?}.", cache_path);
55
56    // Define the fetch definitions for the aligner.
57    let mut fetches = Vec::new();
58
59    // Read the FASTA files and prepare a hashmap of k-mers to search for in the other files.
60    let mut kmer_set = HashSet::new();
61    for fasta_url in fasta_urls {
62        let reader = Reader::from_file(fasta_url.to_file_path().unwrap()).unwrap();
63
64        let mut reads = Vec::new();
65        for record in reader.records().flatten() {
66            let fw_seq = record.seq();
67            let rl_seq = skydive::utils::homopolymer_compressed(fw_seq);
68
69            let kmers = rl_seq
70                .windows(kmer_size)
71                .map(skydive::utils::canonicalize_kmer)
72                .collect::<HashSet<Vec<u8>>>();
73
74            kmer_set.extend(kmers);
75
76            reads.push(fw_seq.to_vec());
77        }
78
79        // Set fetches with the input loci or the detected relevant loci.
80        if search_option == SearchOption::Contig || search_option == SearchOption::ContigAndInterval {
81            match populate_fetches(&search_option, &loci, &ref_path, &reads, &mut fetches) {
82                Ok(_) => {},
83                Err(e) => {
84                    skydive::elog!("Error: {}", e);
85                    std::process::exit(1);
86                }
87            }
88            print_fetches_info(&fetches);
89
90            } else {
91                assert!(
92                    (search_option == SearchOption::All || search_option == SearchOption::Unmapped) && loci.is_none(),
93                    "Assertion failed: search_option is 'All' or 'Unmapped' and loci is NOT None"
94                );
95            }
96    }
97
98
99    // Read the CRAM files and search for the k-mers in each read.
100
101    let mut all_records = Vec::new();
102    for seq_url in seq_urls {
103        let mut reader = skydive::stage::open_bam(&seq_url).unwrap();
104
105        let progress_bar =
106            skydive::utils::default_unbounded_progress_bar("Searching for similar reads (0 found)");
107
108        // Create some thread-safe counters.
109        let found_items = Arc::new(AtomicUsize::new(0));
110        let processed_items = Arc::new(AtomicUsize::new(0));
111        let progress_bar = Arc::new(progress_bar);
112
113        const UPDATE_FREQUENCY: usize = 1_000_000;
114
115        // Create an immutable map of tid to chromosome name
116        let tid_to_chrom: HashMap<i32, String> = reader
117            .header()
118            .target_names()
119            .iter()
120            .enumerate()
121            .map(|(tid, &name)| (tid as i32, String::from_utf8_lossy(name).into_owned()))
122            .collect::<HashMap<_, _>>();
123
124        if search_option == SearchOption::Contig || search_option == SearchOption::ContigAndInterval {
125            // Use the fetches to retrieve the records.
126            retrieve_records(
127                &mut reader,
128                Some(&fetches),
129                &search_option,
130                kmer_size,
131                min_kmers_pct,
132                &kmer_set,
133                &tid_to_chrom,
134                &progress_bar,
135                &found_items,
136                &processed_items,
137                &mut all_records,
138            );
139        } else {
140            // Retrieve all or unmapped reads.
141            retrieve_records(
142                &mut reader,
143                None,
144                &search_option,
145                kmer_size,
146                min_kmers_pct,
147                &kmer_set,
148                &tid_to_chrom,
149                &progress_bar,
150                &found_items,
151                &processed_items,
152                &mut all_records,
153            );
154        }
155
156        let file = File::create(output).expect("Could not open file for writing.");
157        let mut writer: Box<dyn Write> = if output.to_str().unwrap().ends_with(".gz") {
158            Box::new(GzEncoder::new(file, Compression::default()))
159        } else {
160            Box::new(file)
161        };
162
163        for record in all_records.iter() {
164            let fw_seq = record.seq().as_bytes();
165            let rc_seq = fw_seq.reverse_complement();
166
167            writeln!(writer, ">read_{}_{}_{}_{}", String::from_utf8_lossy(&record.qname()), tid_to_chrom.get(&record.tid()).unwrap_or(&"*".to_string()), record.reference_start(), record.reference_end()).expect("Could not write to file");
168            writeln!(
169                writer,
170                "{}",
171                if record.is_reverse() { String::from_utf8_lossy(&rc_seq) } else { String::from_utf8_lossy(&fw_seq) }
172            )
173                .expect("Could not write to file");
174        }
175
176        if !all_records.is_empty() {
177            skydive::elog!(
178                "Wrote {} reads to {}.",
179                all_records.len().to_formatted_string(&Locale::en),
180                output.to_str().unwrap()
181            );
182        } else {
183            skydive::elog!("No reads were found in the CRAM files. Aborting.");
184            std::process::exit(1);
185        }
186    }
187}

195/// This function will populate the fetches with the input loci or the detected relevant loci.
196/// Function should only be called if the search option is set to 'Contig' or 'ContigAndInterval'.
197///
198/// # Arguments
199/// - `search_option` - The search option.
200/// - `loci` - The loci to search for.
201/// - `ref_path` - The path to the reference genome.
202/// - `reads` - The reads to map.
203/// - `fetches` - The fetches to populate.
204///
205/// # Panics
206/// If the search option is set to 'Contig' or 'ContigAndInterval' and no reference genome is provided.
207fn populate_fetches(
208    search_option: &SearchOption,
209    loci: &Option<Vec<String>>,
210    ref_path: &Option<PathBuf>,
211    reads: &Vec<Vec<u8>>,
212    fetches: &mut Vec<(String, Interval<i32>)>,
213) -> Result<(), String>{
214    if !matches!(search_option, SearchOption::Contig | SearchOption::ContigAndInterval) {
215        return Err("Function should only be called if the search option is 'Contig' or 'ContigAndInterval'".to_string());
216    }
217
218    if let Some(loci) = loci {
219        skydive::elog!("Setting loci from input...");
220        let loci = skydive::parse::parse_loci(loci, 0);
221        for (contig, start, stop, _) in loci {
222            match Interval::new(start as i32..stop as i32) {
223                Ok(interval) => fetches.push((contig, interval)),
224                Err(e) => {
225                    skydive::elog!("Invalid interval for contig {}: {:?}, start: {}, stop: {}. Error: {}", contig, start..stop, start, stop, e);
226                    return Err(format!("Error creating interval for contig {}: {:?}", contig, e));
227                }
228            }
229        }
230    } else if let Some(ref ref_path) = ref_path {
231        skydive::elog!("Searching for relevant loci in reference genome...");
232        detect_relevant_loci(ref_path, reads, fetches);
233        fetches.push(("*".to_string(), Interval::new(0..1).unwrap()));
234    } else {
235        skydive::elog!("No reference genome provided but search option is set to 'Contig' or 'ContigAndInterval'. Aborting.");
236        return Err("No reference genome provided.".to_string());
237    }
238
239    Ok(())
240}

242/// This function will print the fetches' info.
243/// # Arguments
244/// - `fetches` - The fetches to print.
245///
246/// # Panics
247///
248/// If the total length of the fetches cannot be calculated.
249fn print_fetches_info(fetches: &[(String, Interval<i32>)]) {
250    let total_length = fetches.iter().map(|(_, interval)| interval.end - interval.start).sum::<i32>();
251    skydive::elog!(
252        " -- will search unaligned reads and {} bases in {} contigs.",
253        total_length.to_formatted_string(&Locale::en),
254        fetches.len().to_formatted_string(&Locale::en)
255    );
256
257    for (contig, interval) in fetches {
258        skydive::elog!(" -- {}:{:?}", contig, interval);
259    }
260}

268/// This function detects the relevant loci in the reference genome.
269///
270/// # Arguments
271///
272/// * `ref_path` - The path to the reference genome.
273/// * `reads` - The reads to map.
274/// * `fetches` - The fetches to populate.
275///
276/// # Panics
277///
278/// If no mappings are found for the provided sequences.
279fn detect_relevant_loci(ref_path: &PathBuf, reads: &Vec<Vec<u8>>, fetches: &mut Vec<(String, Interval<i32>)>) {
280    let lr_aligner = initialize_aligner(ref_path, false);
281
282    let lr_mappings = map_sequences(&lr_aligner, &reads, false);
283
284    for lr_mapping in &lr_mappings {
285        skydive::elog!("lr mapping {:?}:{}-{}", lr_mapping.target_name, lr_mapping.target_start, lr_mapping.target_end);
286    }
287
288    let sr_aligner = initialize_aligner(ref_path, true);
289
290    let sr_mappings = map_sequences(&sr_aligner, &reads, true);
291
292    let mut contig_lengths = HashMap::new();
293    let loci = collect_loci(&lr_mappings, &sr_mappings, &mut contig_lengths);
294
295    // First pass - put all the intervals into a tree.
296    let mut trees = BTreeMap::new();
297    populate_interval_trees(&mut trees, &loci, &contig_lengths);
298
299    // Second pass
300    merge_overlapping_intervals(&mut trees, &loci, &contig_lengths);
301
302    let sub_fetches: HashSet<_> = trees.iter()
303        .flat_map(|(contig_name, tree)| {
304            tree.find(Interval::new(0..i32::MAX).unwrap())
305                .map(move |entry| (contig_name.clone(), entry.interval().clone()))
306        })
307        .collect();
308
309    fetches.extend(sub_fetches.into_iter());
310}

312/// This function initializes the aligner.
313///
314/// # Arguments
315///
316/// * `ref_path` - The path to the reference genome.
317/// * `is_sr` - Whether the reads are short reads or not.
318///
319/// # Returns
320///
321/// An aligner object.
322///
323/// # Panics
324///
325/// If the index cannot be built.
326fn initialize_aligner(ref_path: &PathBuf, is_sr: bool) -> Aligner<Built> {
327    if is_sr {
328        Aligner::builder()
329            .sr()
330            .with_sam_hit_only()
331            .with_index(ref_path, None)
332            .expect("Unable to build index")
333    } else {
334        Aligner::builder()
335            .map_hifi()
336            .with_sam_hit_only()
337            .with_index(ref_path, None)
338            .expect("Unable to build index")
339    }
340}

342/// This function maps the sequences to the reference genome using the aligner.
343///
344/// # Arguments
345///
346/// * `aligner` - The aligner object to use for mapping.
347/// * `reads` - The reads to map.
348/// * `is_sr` - Whether the reads are short reads or not.
349///
350/// # Returns
351///
352/// A vector of mappings.
353///
354/// # Panics
355///
356/// If no mappings are found for the provided sequences.
357fn map_sequences(aligner: &Aligner<Built>, reads: &[Vec<u8>], is_sr: bool) -> Vec<Mapping> {
358    let mappings: Vec<Mapping> = reads.par_iter().flat_map(|seq| {
359        if is_sr {
360            seq.windows(150).flat_map(|window| aligner.map(window, false, false, None, None, None).unwrap_or_else(|_| vec![])).collect::<Vec<_>>()
361        } else {
362            aligner.map(seq, false, false, None, None, None).unwrap_or_else(|_| vec![])
363        }
364    }).collect();
365
366    if mappings.is_empty() {
367        panic!("No mappings found for the provided sequences.");
368    }
369
370    mappings
371}

373/// This function collects the loci from the mappings.
374///
375/// # Arguments
376///
377/// * `lr_mappings` - The long read mappings.
378/// * `sr_mappings` - The short read mappings.
379///
380/// # Returns
381///
382/// A hashset of loci.
383///
384/// # Panics
385///
386/// If the target name is not found in the mappings.
387fn collect_loci(
388    lr_mappings: &[Mapping],
389    sr_mappings: &[Mapping],
390    contig_lengths: &mut HashMap<String, i32>,
391) -> HashSet<(Vec<u8>, i32, i32)> {
392    lr_mappings.iter().chain(sr_mappings.iter()).filter_map(|m| {
393        if let Some(target_name) = &m.target_name {
394            contig_lengths.insert(target_name.to_string(), m.target_len);
395            Some((target_name.to_string().as_bytes().to_vec(), m.target_start, m.target_end))
396        } else {
397            None
398        }
399    }).collect::<HashSet<_>>()
400}

402/// This function populates the interval trees with the loci.
403///
404/// # Arguments
405///
406/// * `trees` - The interval trees to populate.
407/// * `loci` - The loci to insert into the trees.
408/// * `contig_lengths` - The lengths of the contigs.
409///
410/// # Panics
411///
412/// If the contig name is not found in the contig lengths.
413fn populate_interval_trees(
414    trees: &mut BTreeMap<String, IntervalTree<i32, ()>>,
415    loci: &HashSet<(Vec<u8>, i32, i32)>,
416    contig_lengths: &HashMap<String, i32>
417) {
418    for (seq, start, end) in loci {
419        let contig_name = String::from_utf8_lossy(&seq).to_string();
420        let contig_length = contig_lengths.get(&contig_name).unwrap();
421
422        if !trees.contains_key(&contig_name) {
423            trees.insert(contig_name.clone(), IntervalTree::new());
424        }
425
426        let interval = Interval::new(start.saturating_sub(50000).max(0)..end.saturating_add(50000).min(*contig_length)).unwrap();
427        trees.get_mut(&contig_name).unwrap().insert(interval, ());
428    }
429}

431/// This function merges overlapping intervals.
432///
433/// # Arguments
434///
435/// * `trees` - The interval trees to merge.
436/// * `loci` - The loci to merge.
437/// * `contig_lengths` - The lengths of the contigs.
438///
439/// # Panics
440///
441/// If the contig name is not found in the trees.
442fn merge_overlapping_intervals(
443    trees: &mut BTreeMap<String, IntervalTree<i32, ()>>,
444    loci: &HashSet<(Vec<u8>, i32, i32)>,
445    contig_lengths: &HashMap<String, i32>
446) {
447    for (seq, start, end) in loci {
448        let contig_name = String::from_utf8_lossy(&seq).to_string();
449        let contig_length = contig_lengths.get(&contig_name).unwrap();
450        let tree = trees.get_mut(&contig_name).unwrap();
451
452        let current_interval = Interval::new(start.saturating_sub(50000).max(0)..end.saturating_add(50000).min(*contig_length)).unwrap();
453        let overlaps = tree.find(current_interval.clone()).collect::<Vec<_>>();
454
455        let mut merged_intervals = IntervalTree::new();
456
457        if !overlaps.is_empty() {
458            let min_start = overlaps.iter().map(|o| o.interval().start).min().unwrap().min(*start);
459            let max_end = overlaps.iter().map(|o| o.interval().end).max().unwrap().max(*end);
460            merged_intervals.insert(Interval::new(min_start..max_end).unwrap(), ());
461        } else {
462            merged_intervals.insert(current_interval, ());
463        }
464
465        trees.insert(contig_name.clone(), merged_intervals);
466    }
467}

474/// This function retrieves the records from the reader.
475///
476/// # Arguments
477/// - `fetches` - The fetches to process.
478/// - `reader` - The reader to use for fetching reads.
479/// - `kmer_size` - The size of the k-mers to search for.
480/// - `min_kmers_pct` - The minimum percentage of k-mers to search for.
481/// - `search_option` - Whether to search all reads or not.
482/// - `kmer_set` - The set of k-mers to search for.
483/// - `tid_to_chrom` - The mapping of TID to chromosome name.
484/// - `progress_bar` - The progress bar to update.
485/// - `found_items` - The number of found items.
486/// - `processed_items` - The number of processed items.
487/// - `all_records` - The vector of all records.
488///
489fn retrieve_records(
490    reader: &mut IndexedReader,
491    fetches: Option<&[(String, Interval<i32>)]>,
492    search_option: &SearchOption,
493    kmer_size: usize,
494    min_kmers_pct: usize,
495    kmer_set: &HashSet<Vec<u8>>,
496    tid_to_chrom: &HashMap<i32, String>,
497    progress_bar: &Arc<ProgressBar>,
498    found_items: &Arc<AtomicUsize>,
499    processed_items: &Arc<AtomicUsize>,
500    all_records: &mut Vec<BamRecord>,
501) {
502    const UPDATE_FREQUENCY: usize = 1_000_000;
503
504    if let Some(fetches) = fetches {
505        let fetches_updated = prepare_fetches(fetches, search_option);
506
507        for (contig, interval) in fetches_updated {
508            let fetch_definition = create_fetch_definition(search_option, Some(&contig), Some(interval));
509            reader.fetch(fetch_definition).expect("Failed to fetch reads");
510
511            let records = filter_and_collect_records(
512                reader,
513                kmer_size,
514                min_kmers_pct,
515                kmer_set,
516                tid_to_chrom,
517                progress_bar,
518                found_items,
519                processed_items,
520            );
521
522            all_records.extend(records);
523        }
524    } else {
525        let fetch_definition = create_fetch_definition(search_option, None, None);
526        reader.fetch(fetch_definition).expect("Failed to fetch reads");
527
528        let records = filter_and_collect_records(
529            reader,
530            kmer_size,
531            min_kmers_pct,
532            kmer_set,
533            tid_to_chrom,
534            progress_bar,
535            found_items,
536            processed_items,
537        );
538
539        all_records.extend(records);
540    }
541
542    finalize_progress(progress_bar, found_items, processed_items);
543}

545/// This function processes all or unmapped reads.
546/// # Arguments
547/// - `reader` - The reader to use for fetching reads.
548/// - `kmer_size` - The size of the k-mers to search for.
549/// - `min_kmers_pct` - The minimum percentage of k-mers to search for.
550/// - `kmer_set` - The set of k-mers to search for.
551/// - `tid_to_chrom` - The mapping of TID to chromosome name.
552/// - `progress_bar` - The progress bar to update.
553/// - `found_items` - The number of found items.
554/// - `processed_items` - The number of processed items.
555///
556/// # Panics
557/// If the search option is not 'All' or 'Unmapped'.
558///
559/// # Returns
560/// A vector of all records.
561fn filter_and_collect_records(
562    reader: &mut IndexedReader,
563    kmer_size: usize,
564    min_kmers_pct: usize,
565    kmer_set: &HashSet<Vec<u8>>,
566    tid_to_chrom: &HashMap<i32, String>,
567    progress_bar: &Arc<ProgressBar>,
568    found_items: &Arc<AtomicUsize>,
569    processed_items: &Arc<AtomicUsize>,
570) -> Vec<BamRecord> {
571    const UPDATE_FREQUENCY: usize = 1_000_000;
572
573    reader
574        .records()
575        .par_bridge()
576        .flat_map(|record| record.ok())
577        .filter_map(|read| {
578            update_processed_progress(processed_items, UPDATE_FREQUENCY, tid_to_chrom, &read, &progress_bar);
579
580            if is_valid_read(&read, kmer_size, min_kmers_pct, kmer_set) {
581                update_found_progress(found_items, UPDATE_FREQUENCY, tid_to_chrom, &read, &progress_bar);
582                Some(read)
583            } else {
584                None
585            }
586        })
587        .collect()
588}

590/// This function prepares the fetches for searching.
591/// If the search option is 'Contig', it will extract the unique contigs from the fetches and add a dummy interval.
592/// If the search option is 'ContigAndInterval', it will return the fetches as is.
593/// In both cases, the fetches will be returned as a vector of contigs and intervals.
594///
595/// # Arguments
596/// - `fetches` - The fetches to prepare.
597/// - `search_option` - Whether to search all reads or not.
598///
599/// # Returns
600/// A vector of fetches to search.
601///
602/// # Panics
603/// If the contig is not valid.
604fn prepare_fetches(
605    fetches: &[(String, Interval<i32>)], search_option: &SearchOption
606) -> Vec<(String, Interval<i32>)> {
607    if *search_option == SearchOption::Contig {
608        let uniq_contigs: HashSet<_> = fetches.iter().map(|(contig, _)| contig.clone()).collect();
609        uniq_contigs.into_iter().map(|contig| (contig, Interval::new(0..1).unwrap())).collect()
610    } else if *search_option == SearchOption::ContigAndInterval {
611        fetches.iter()
612            .map(|(contig, interval)| (contig.clone(), interval.clone()))
613            .collect()
614    } else {
615        panic!("Invalid search option");
616    }
617}

619/// This function creates a fetch definition.
620///
621/// # Arguments
622/// - `search_option` - Whether to search all reads or not.
623/// - `contig` - The contig to fetch.
624/// - `interval` - The interval to fetch.
625///
626/// # Returns
627/// A fetch definition.
628fn create_fetch_definition<'a>(
629    search_option: &'a SearchOption,
630    contig: Option<&'a String>,
631    interval: Option<Interval<i32>>,
632) -> FetchDefinition<'a> {
633    match (contig, search_option) {
634        (None, SearchOption::All) => FetchDefinition::All,
635        (None, SearchOption::Unmapped) => FetchDefinition::Unmapped,
636        (Some(contig), SearchOption::Contig) if contig != "*" => {
637            FetchDefinition::String(contig.as_bytes())
638        },
639        (Some(contig), SearchOption::ContigAndInterval) if contig != "*" => {
640            let interval = interval.unwrap_or_else(|| Interval::new(0..1).unwrap());
641            FetchDefinition::RegionString(contig.as_bytes(), interval.start as i64, interval.end as i64)
642        },
643        (Some(contig), _) if contig == "*" => FetchDefinition::Unmapped,
644        _ => panic!(
645            "Invalid fetch definition settings. \
646                If you are using the 'Contig' or 'ContigAndInterval' search options, \
647                you must provide a contig name. \
648                If you are using the 'All' or 'Unmapped' search options, \
649                you must NOT provide a contig name. \
650             search option: '{:?}' , contig: '{:?}', interval: '{:?}'",
651            search_option,
652            contig.as_deref().unwrap_or(&"None".to_string()),
653            interval.map(|i| format!("{:?}", i)).unwrap_or_else(|| "None".to_string())
654        ),
655    }
656}

658/// This function updates the processed progress.
659///
660/// # Arguments
661/// - `processed_items` - The number of processed items.
662/// - `update_freq` - The update frequency.
663/// - `tid_to_chrom` - The mapping of TID to chromosome name.
664/// - `read` - The read to process.
665/// - `progress_bar` - The progress bar to update.
666///
667/// # Panics
668/// If the chromosome name is not found in the mapping.
669fn update_processed_progress(processed_items: &Arc<AtomicUsize>, update_freq: usize, tid_to_chrom: &HashMap<i32, String>, read: &BamRecord, progress_bar: &Arc<ProgressBar>) {
670    let current_processed = processed_items.fetch_add(1, Ordering::Relaxed);
671    if current_processed % update_freq == 0 {
672        let unknown_chrom = "Unknown".to_string();
673        let chrom_name = tid_to_chrom.get(&read.tid()).unwrap_or(&unknown_chrom);
674        progress_bar.set_message(format!(
675            "Searching for similar reads ({} processed, most recent at {}:{})",
676            current_processed.to_formatted_string(&Locale::en),
677            chrom_name,
678            read.reference_start().to_formatted_string(&Locale::en)
679        ));
680        progress_bar.inc(update_freq as u64);
681    }
682}

709/// This function checks if a read is valid.
710///
711/// # Arguments
712/// - `read` - The read to check.
713/// - `kmer_size` - The size of the k-mers to search for.
714/// - `min_kmers_pct` - The minimum percentage of k-mers to search for.
715/// - `kmer_set` - The set of k-mers to search for.
716///
717/// # Returns
718/// Whether the read is valid or not.
719fn is_valid_read(read: &BamRecord, kmer_size: usize, min_kmers_pct: usize, kmer_set: &HashSet<Vec<u8>>) -> bool {
720    let read_seq = read.seq().as_bytes();
721    let rl_seq = skydive::utils::homopolymer_compressed(&read_seq);
722    let num_kmers = rl_seq.par_windows(kmer_size)
723        .filter(|kmer| kmer_set.contains(&skydive::utils::canonicalize_kmer(kmer)))
724        .count();
725    (num_kmers as f32 / rl_seq.len() as f32) > (min_kmers_pct as f32 / 100.0)
726}

728/// This function finalizes the progress.
729///
730/// # Arguments
731/// - `progress_bar` - The progress bar to finalize.
732/// - `found_items` - The number of found items.
733/// - `processed_items` - The number of processed items.
734fn finalize_progress(progress_bar: &Arc<ProgressBar>, found_items: &Arc<AtomicUsize>, processed_items: &Arc<AtomicUsize>) {
735    let final_found = found_items.load(Ordering::Relaxed);
736    let final_processed = processed_items.load(Ordering::Relaxed);
737    progress_bar.set_message(format!(
738        "Found {} reads in {} reads total.",
739        final_found.to_formatted_string(&Locale::en),
740        final_processed.to_formatted_string(&Locale::en)
741    ));
742    progress_bar.finish();
743}
