38pub fn start(
39    output: &PathBuf,
40    kmer_size: usize,
41    min_kmers_pct: usize,
42    search_option: SearchOption,
43    ref_path: Option<PathBuf>,
44    loci: Option<Vec<String>>,
45    fastq_paths: &Vec<PathBuf>,
46    seq_paths: &Vec<PathBuf>,
47) {
48    let fastq_urls = skydive::parse::parse_file_names(fastq_paths);
49    let seq_urls = skydive::parse::parse_file_names(seq_paths);
50
51    // Get the system's temporary directory path
52    let cache_path = std::env::temp_dir();
53    skydive::elog!("Intermediate data will be stored at {:?}.", cache_path);
54
55    // Define the fetch definitions for the aligner.
56    let mut fetches = Vec::new();
57
58    // Read the FASTA files and prepare a hashmap of k-mers to search for in the other files.
59    let mut kmer_set = HashSet::new();
60    for fastq_url in fastq_urls {
61        let reader = Reader::from_file(fastq_url.to_file_path().unwrap()).unwrap();
62
63        let mut reads = Vec::new();
64        for record in reader.records().flatten() {
65            let fw_seq = record.seq();
66            let rl_seq = skydive::utils::homopolymer_compressed(fw_seq);
67
68            let kmers = rl_seq
69                .windows(kmer_size)
70                .map(skydive::utils::canonicalize_kmer)
71                .collect::<HashSet<Vec<u8>>>();
72
73            kmer_set.extend(kmers);
74
75            reads.push(fw_seq.to_vec());
76        }
77
78        // Set fetches with the input loci or the detected relevant loci.
79        if search_option == SearchOption::Contig || search_option == SearchOption::ContigAndInterval
80        {
81            match populate_fetches(&search_option, &loci, &ref_path, &reads, &mut fetches) {
82                Ok(_) => {}
83                Err(e) => {
84                    skydive::elog!("Error: {}", e);
85                    std::process::exit(1);
86                }
87            }
88            print_fetches_info(&fetches);
89        } else {
90            assert!(
91                (search_option == SearchOption::All || search_option == SearchOption::Unmapped)
92                    && loci.is_none(),
93                "Assertion failed: search_option is 'All' or 'Unmapped' and loci is NOT None"
94            );
95        }
96    }
97
98    // Read the CRAM files and search for the k-mers in each read.
99
100    let mut all_records = Vec::new();
101    for seq_url in seq_urls {
102        let mut reader = skydive::stage::open_bam(&seq_url).unwrap();
103
104        let progress_bar =
105            skydive::utils::default_unbounded_progress_bar("Searching for similar reads (0 found)");
106
107        // Create some thread-safe counters.
108        let found_items = Arc::new(AtomicUsize::new(0));
109        let processed_items = Arc::new(AtomicUsize::new(0));
110        let progress_bar = Arc::new(progress_bar);
111
112        const UPDATE_FREQUENCY: usize = 1_000_000;
113
114        // Create an immutable map of tid to chromosome name
115        let tid_to_chrom: HashMap<i32, String> = reader
116            .header()
117            .target_names()
118            .iter()
119            .enumerate()
120            .map(|(tid, &name)| (tid as i32, String::from_utf8_lossy(name).into_owned()))
121            .collect::<HashMap<_, _>>();
122
123        if search_option == SearchOption::Contig || search_option == SearchOption::ContigAndInterval
124        {
125            // Use the fetches to retrieve the records.
126            retrieve_records(
127                &mut reader,
128                Some(&fetches),
129                &search_option,

139        } else {

141            retrieve_records(
142                &mut reader,

144                &search_option,
145                kmer_size,
146                min_kmers_pct,
147                &kmer_set,
148                &tid_to_chrom,

154        }

159            Box::new(GzEncoder::new(file, Compression::default()))
160        } else {
161            Box::new(file)
162        };
163        */
164
165        let mut buf_writer = BufWriter::new(File::create(output).unwrap());
166        let mut fastq_writer = fastq::Writer::new(&mut buf_writer);

186                writer,

190                record.reference_start(),

192            )

194            writeln!(

225/// This function will populate the fetches with the input loci or the detected relevant loci.
226/// Function should only be called if the search option is set to 'Contig' or 'ContigAndInterval'.
227///
228/// # Arguments
229/// - `search_option` - The search option.
230/// - `loci` - The loci to search for.
231/// - `ref_path` - The path to the reference genome.
232/// - `reads` - The reads to map.
233/// - `fetches` - The fetches to populate.
234///
235/// # Panics
236/// If the search option is set to 'Contig' or 'ContigAndInterval' and no reference genome is provided.
237fn populate_fetches(
238    search_option: &SearchOption,
239    loci: &Option<Vec<String>>,
240    ref_path: &Option<PathBuf>,
241    reads: &Vec<Vec<u8>>,
242    fetches: &mut Vec<(String, Interval<i32>)>,
243) -> Result<(), String> {
244    if !matches!(
245        search_option,
246        SearchOption::Contig | SearchOption::ContigAndInterval
247    ) {
248        return Err("Function should only be called if the search option is 'Contig' or 'ContigAndInterval'".to_string());
249    }
250
251    if let Some(loci) = loci {
252        skydive::elog!("Setting loci from input...");
253        let loci = skydive::parse::parse_loci(loci, 0);
254        for (contig, start, stop, _) in loci {
255            match Interval::new(start as i32..stop as i32) {
256                Ok(interval) => fetches.push((contig, interval)),
257                Err(e) => {
258                    skydive::elog!(
259                        "Invalid interval for contig {}: {:?}, start: {}, stop: {}. Error: {}",
260                        contig,
261                        start..stop,
262                        start,
263                        stop,
264                        e
265                    );
266                    return Err(format!(
267                        "Error creating interval for contig {}: {:?}",
268                        contig, e
269                    ));
270                }
271            }
272        }
273    } else if let Some(ref ref_path) = ref_path {
274        skydive::elog!("Searching for relevant loci in reference genome...");
275        detect_relevant_loci(ref_path, reads, fetches);
276        fetches.push(("*".to_string(), Interval::new(0..1).unwrap()));
277    } else {
278        skydive::elog!("No reference genome provided but search option is set to 'Contig' or 'ContigAndInterval'. Aborting.");
279        return Err("No reference genome provided.".to_string());
280    }
281
282    Ok(())
283}

285/// This function will print the fetches' info.
286/// # Arguments
287/// - `fetches` - The fetches to print.
288///
289/// # Panics
290///
291/// If the total length of the fetches cannot be calculated.
292fn print_fetches_info(fetches: &[(String, Interval<i32>)]) {
293    let total_length = fetches
294        .iter()
295        .map(|(_, interval)| interval.end - interval.start)
296        .sum::<i32>();
297    skydive::elog!(
298        " -- will search unaligned reads and {} bases in {} contigs.",
299        total_length.to_formatted_string(&Locale::en),
300        fetches.len().to_formatted_string(&Locale::en)
301    );
302
303    for (contig, interval) in fetches {
304        skydive::elog!(" -- {}:{:?}", contig, interval);
305    }
306}

313/// This function detects the relevant loci in the reference genome.
314///
315/// # Arguments
316///
317/// * `ref_path` - The path to the reference genome.
318/// * `reads` - The reads to map.
319/// * `fetches` - The fetches to populate.
320///
321/// # Panics
322///
323/// If no mappings are found for the provided sequences.
324fn detect_relevant_loci(
325    ref_path: &PathBuf,
326    reads: &Vec<Vec<u8>>,
327    fetches: &mut Vec<(String, Interval<i32>)>,
328) {
329    let lr_aligner = initialize_aligner(ref_path, false);
330
331    let lr_mappings = map_sequences(&lr_aligner, &reads, false);
332
333    for lr_mapping in &lr_mappings {
334        skydive::elog!(
335            "lr mapping {:?}:{}-{}",
336            lr_mapping.target_name,
337            lr_mapping.target_start,
338            lr_mapping.target_end
339        );
340    }
341
342    let sr_aligner = initialize_aligner(ref_path, true);
343
344    let sr_mappings = map_sequences(&sr_aligner, &reads, true);
345
346    let mut contig_lengths = HashMap::new();
347    let loci = collect_loci(&lr_mappings, &sr_mappings, &mut contig_lengths);
348
349    // First pass - put all the intervals into a tree.
350    let mut trees = BTreeMap::new();
351    populate_interval_trees(&mut trees, &loci, &contig_lengths);
352
353    // Second pass
354    merge_overlapping_intervals(&mut trees, &loci, &contig_lengths);
355
356    let sub_fetches: HashSet<_> = trees
357        .iter()
358        .flat_map(|(contig_name, tree)| {
359            tree.find(Interval::new(0..i32::MAX).unwrap())
360                .map(move |entry| (contig_name.clone(), entry.interval().clone()))
361        })
362        .collect();
363
364    fetches.extend(sub_fetches.into_iter());
365}

367/// This function initializes the aligner.
368///
369/// # Arguments
370///
371/// * `ref_path` - The path to the reference genome.
372/// * `is_sr` - Whether the reads are short reads or not.
373///
374/// # Returns
375///
376/// An aligner object.
377///
378/// # Panics
379///
380/// If the index cannot be built.
381fn initialize_aligner(ref_path: &PathBuf, is_sr: bool) -> Aligner<Built> {
382    if is_sr {
383        Aligner::builder()
384            .sr()
385            .with_sam_hit_only()
386            .with_index(ref_path, None)
387            .expect("Unable to build index")
388    } else {
389        Aligner::builder()
390            .map_hifi()
391            .with_sam_hit_only()
392            .with_index(ref_path, None)
393            .expect("Unable to build index")
394    }
395}

397/// This function maps the sequences to the reference genome using the aligner.
398///
399/// # Arguments
400///
401/// * `aligner` - The aligner object to use for mapping.
402/// * `reads` - The reads to map.
403/// * `is_sr` - Whether the reads are short reads or not.
404///
405/// # Returns
406///
407/// A vector of mappings.
408///
409/// # Panics
410///
411/// If no mappings are found for the provided sequences.
412fn map_sequences(aligner: &Aligner<Built>, reads: &[Vec<u8>], is_sr: bool) -> Vec<Mapping> {
413    let mappings: Vec<Mapping> = reads
414        .par_iter()
415        .flat_map(|seq| {
416            if is_sr {
417                seq.windows(150)
418                    .flat_map(|window| {
419                        aligner
420                            .map(window, false, false, None, None, None)
421                            .unwrap_or_else(|_| vec![])
422                    })
423                    .collect::<Vec<_>>()
424            } else {
425                aligner
426                    .map(seq, false, false, None, None, None)
427                    .unwrap_or_else(|_| vec![])
428            }
429        })
430        .collect();
431
432    if mappings.is_empty() {
433        panic!("No mappings found for the provided sequences.");
434    }
435
436    mappings
437}

439/// This function collects the loci from the mappings.
440///
441/// # Arguments
442///
443/// * `lr_mappings` - The long read mappings.
444/// * `sr_mappings` - The short read mappings.
445///
446/// # Returns
447///
448/// A hashset of loci.
449///
450/// # Panics
451///
452/// If the target name is not found in the mappings.
453fn collect_loci(
454    lr_mappings: &[Mapping],
455    sr_mappings: &[Mapping],
456    contig_lengths: &mut HashMap<String, i32>,
457) -> HashSet<(Vec<u8>, i32, i32)> {
458    lr_mappings
459        .iter()
460        .chain(sr_mappings.iter())
461        .filter_map(|m| {
462            if let Some(target_name) = &m.target_name {
463                contig_lengths.insert(target_name.to_string(), m.target_len);
464                Some((
465                    target_name.to_string().as_bytes().to_vec(),
466                    m.target_start,
467                    m.target_end,
468                ))
469            } else {
470                None
471            }
472        })
473        .collect::<HashSet<_>>()
474}

476/// This function populates the interval trees with the loci.
477///
478/// # Arguments
479///
480/// * `trees` - The interval trees to populate.
481/// * `loci` - The loci to insert into the trees.
482/// * `contig_lengths` - The lengths of the contigs.
483///
484/// # Panics
485///
486/// If the contig name is not found in the contig lengths.
487fn populate_interval_trees(
488    trees: &mut BTreeMap<String, IntervalTree<i32, ()>>,
489    loci: &HashSet<(Vec<u8>, i32, i32)>,
490    contig_lengths: &HashMap<String, i32>,
491) {
492    for (seq, start, end) in loci {
493        let contig_name = String::from_utf8_lossy(&seq).to_string();
494        let contig_length = contig_lengths.get(&contig_name).unwrap();
495
496        if !trees.contains_key(&contig_name) {
497            trees.insert(contig_name.clone(), IntervalTree::new());
498        }
499
500        let interval = Interval::new(
501            start.saturating_sub(50000).max(0)..end.saturating_add(50000).min(*contig_length),
502        )
503        .unwrap();
504        trees.get_mut(&contig_name).unwrap().insert(interval, ());
505    }
506}

508/// This function merges overlapping intervals.
509///
510/// # Arguments
511///
512/// * `trees` - The interval trees to merge.
513/// * `loci` - The loci to merge.
514/// * `contig_lengths` - The lengths of the contigs.
515///
516/// # Panics
517///
518/// If the contig name is not found in the trees.
519fn merge_overlapping_intervals(
520    trees: &mut BTreeMap<String, IntervalTree<i32, ()>>,
521    loci: &HashSet<(Vec<u8>, i32, i32)>,
522    contig_lengths: &HashMap<String, i32>,
523) {
524    for (seq, start, end) in loci {
525        let contig_name = String::from_utf8_lossy(&seq).to_string();
526        let contig_length = contig_lengths.get(&contig_name).unwrap();
527        let tree = trees.get_mut(&contig_name).unwrap();
528
529        let current_interval = Interval::new(
530            start.saturating_sub(50000).max(0)..end.saturating_add(50000).min(*contig_length),
531        )
532        .unwrap();
533        let overlaps = tree.find(current_interval.clone()).collect::<Vec<_>>();
534
535        let mut merged_intervals = IntervalTree::new();
536
537        if !overlaps.is_empty() {
538            let min_start = overlaps
539                .iter()
540                .map(|o| o.interval().start)
541                .min()
542                .unwrap()
543                .min(*start);
544            let max_end = overlaps
545                .iter()
546                .map(|o| o.interval().end)
547                .max()
548                .unwrap()
549                .max(*end);
550            merged_intervals.insert(Interval::new(min_start..max_end).unwrap(), ());
551        } else {
552            merged_intervals.insert(current_interval, ());
553        }
554
555        trees.insert(contig_name.clone(), merged_intervals);
556    }
557}

564/// This function retrieves the records from the reader.
565///
566/// # Arguments
567/// - `fetches` - The fetches to process.
568/// - `reader` - The reader to use for fetching reads.
569/// - `kmer_size` - The size of the k-mers to search for.
570/// - `min_kmers_pct` - The minimum percentage of k-mers to search for.
571/// - `search_option` - Whether to search all reads or not.
572/// - `kmer_set` - The set of k-mers to search for.
573/// - `tid_to_chrom` - The mapping of TID to chromosome name.
574/// - `progress_bar` - The progress bar to update.
575/// - `found_items` - The number of found items.
576/// - `processed_items` - The number of processed items.
577/// - `all_records` - The vector of all records.
578///
579fn retrieve_records(
580    reader: &mut IndexedReader,
581    fetches: Option<&[(String, Interval<i32>)]>,
582    search_option: &SearchOption,
583    kmer_size: usize,
584    min_kmers_pct: usize,
585    kmer_set: &HashSet<Vec<u8>>,
586    tid_to_chrom: &HashMap<i32, String>,
587    progress_bar: &Arc<ProgressBar>,
588    found_items: &Arc<AtomicUsize>,
589    processed_items: &Arc<AtomicUsize>,
590    all_records: &mut Vec<BamRecord>,
591) {
592    const UPDATE_FREQUENCY: usize = 1_000_000;
593
594    if let Some(fetches) = fetches {
595        let fetches_updated = prepare_fetches(fetches, search_option);
596
597        for (contig, interval) in fetches_updated {
598            let fetch_definition =
599                create_fetch_definition(search_option, Some(&contig), Some(interval));
600            reader
601                .fetch(fetch_definition)
602                .expect("Failed to fetch reads");
603
604            let records = filter_and_collect_records(
605                reader,
606                kmer_size,
607                min_kmers_pct,
608                kmer_set,
609                tid_to_chrom,
610                progress_bar,
611                found_items,
612                processed_items,
613            );
614
615            all_records.extend(records);
616        }
617    } else {
618        let fetch_definition = create_fetch_definition(search_option, None, None);
619        reader
620            .fetch(fetch_definition)
621            .expect("Failed to fetch reads");
622
623        let records = filter_and_collect_records(
624            reader,
625            kmer_size,
626            min_kmers_pct,
627            kmer_set,
628            tid_to_chrom,
629            progress_bar,
630            found_items,
631            processed_items,
632        );
633
634        all_records.extend(records);
635    }
636
637    finalize_progress(progress_bar, found_items, processed_items);
638}

640/// This function processes all or unmapped reads.
641/// # Arguments
642/// - `reader` - The reader to use for fetching reads.
643/// - `kmer_size` - The size of the k-mers to search for.
644/// - `min_kmers_pct` - The minimum percentage of k-mers to search for.
645/// - `kmer_set` - The set of k-mers to search for.
646/// - `tid_to_chrom` - The mapping of TID to chromosome name.
647/// - `progress_bar` - The progress bar to update.
648/// - `found_items` - The number of found items.
649/// - `processed_items` - The number of processed items.
650///
651/// # Panics
652/// If the search option is not 'All' or 'Unmapped'.
653///
654/// # Returns
655/// A vector of all records.
656fn filter_and_collect_records(
657    reader: &mut IndexedReader,
658    kmer_size: usize,
659    min_kmers_pct: usize,
660    kmer_set: &HashSet<Vec<u8>>,
661    tid_to_chrom: &HashMap<i32, String>,
662    progress_bar: &Arc<ProgressBar>,
663    found_items: &Arc<AtomicUsize>,
664    processed_items: &Arc<AtomicUsize>,
665) -> Vec<BamRecord> {
666    const UPDATE_FREQUENCY: usize = 1_000_000;
667
668    reader
669        .records()
670        .par_bridge()
671        .flat_map(|record| record.ok())
672        .filter_map(|read| {
673            update_processed_progress(
674                processed_items,
675                UPDATE_FREQUENCY,
676                tid_to_chrom,
677                &read,
678                &progress_bar,
679            );
680
681            if is_valid_read(&read, kmer_size, min_kmers_pct, kmer_set) {
682                update_found_progress(
683                    found_items,
684                    UPDATE_FREQUENCY,
685                    tid_to_chrom,
686                    &read,
687                    &progress_bar,
688                );
689                Some(read)
690            } else {
691                None
692            }
693        })
694        .collect()
695}

697/// This function prepares the fetches for searching.
698/// If the search option is 'Contig', it will extract the unique contigs from the fetches and add a dummy interval.
699/// If the search option is 'ContigAndInterval', it will return the fetches as is.
700/// In both cases, the fetches will be returned as a vector of contigs and intervals.
701///
702/// # Arguments
703/// - `fetches` - The fetches to prepare.
704/// - `search_option` - Whether to search all reads or not.
705///
706/// # Returns
707/// A vector of fetches to search.
708///
709/// # Panics
710/// If the contig is not valid.
711fn prepare_fetches(
712    fetches: &[(String, Interval<i32>)],
713    search_option: &SearchOption,
714) -> Vec<(String, Interval<i32>)> {
715    if *search_option == SearchOption::Contig {
716        let uniq_contigs: HashSet<_> = fetches.iter().map(|(contig, _)| contig.clone()).collect();
717        uniq_contigs
718            .into_iter()
719            .map(|contig| (contig, Interval::new(0..1).unwrap()))
720            .collect()
721    } else if *search_option == SearchOption::ContigAndInterval {
722        fetches
723            .iter()
724            .map(|(contig, interval)| (contig.clone(), interval.clone()))
725            .collect()
726    } else {
727        panic!("Invalid search option");
728    }
729}

731/// This function creates a fetch definition.
732///
733/// # Arguments
734/// - `search_option` - Whether to search all reads or not.
735/// - `contig` - The contig to fetch.
736/// - `interval` - The interval to fetch.
737///
738/// # Returns
739/// A fetch definition.
740fn create_fetch_definition<'a>(
741    search_option: &'a SearchOption,
742    contig: Option<&'a String>,
743    interval: Option<Interval<i32>>,
744) -> FetchDefinition<'a> {
745    match (contig, search_option) {
746        (None, SearchOption::All) => FetchDefinition::All,
747        (None, SearchOption::Unmapped) => FetchDefinition::Unmapped,
748        (Some(contig), SearchOption::Contig) if contig != "*" => {
749            FetchDefinition::String(contig.as_bytes())
750        }
751        (Some(contig), SearchOption::ContigAndInterval) if contig != "*" => {
752            let interval = interval.unwrap_or_else(|| Interval::new(0..1).unwrap());
753            FetchDefinition::RegionString(
754                contig.as_bytes(),
755                interval.start as i64,
756                interval.end as i64,
757            )
758        }
759        (Some(contig), _) if contig == "*" => FetchDefinition::Unmapped,
760        _ => panic!(
761            "Invalid fetch definition settings. \
762                If you are using the 'Contig' or 'ContigAndInterval' search options, \
763                you must provide a contig name. \
764                If you are using the 'All' or 'Unmapped' search options, \
765                you must NOT provide a contig name. \
766             search option: '{:?}' , contig: '{:?}', interval: '{:?}'",
767            search_option,
768            contig.as_deref().unwrap_or(&"None".to_string()),
769            interval
770                .map(|i| format!("{:?}", i))
771                .unwrap_or_else(|| "None".to_string())
772        ),
773    }
774}

776/// This function updates the processed progress.
777///
778/// # Arguments
779/// - `processed_items` - The number of processed items.
780/// - `update_freq` - The update frequency.
781/// - `tid_to_chrom` - The mapping of TID to chromosome name.
782/// - `read` - The read to process.
783/// - `progress_bar` - The progress bar to update.
784///
785/// # Panics
786/// If the chromosome name is not found in the mapping.
787fn update_processed_progress(
788    processed_items: &Arc<AtomicUsize>,
789    update_freq: usize,
790    tid_to_chrom: &HashMap<i32, String>,
791    read: &BamRecord,
792    progress_bar: &Arc<ProgressBar>,
793) {
794    let current_processed = processed_items.fetch_add(1, Ordering::Relaxed);
795    if current_processed % update_freq == 0 {
796        let unknown_chrom = "Unknown".to_string();
797        let chrom_name = tid_to_chrom.get(&read.tid()).unwrap_or(&unknown_chrom);
798        progress_bar.set_message(format!(
799            "Searching for similar reads ({} processed, most recent at {}:{})",
800            current_processed.to_formatted_string(&Locale::en),
801            chrom_name,
802            read.reference_start().to_formatted_string(&Locale::en)
803        ));
804        progress_bar.inc(update_freq as u64);
805    }
806}

811/// - `found_items` - The number of found items.
812/// - `update_freq` - The update frequency.

823    read: &BamRecord,

839/// This function checks if a read is valid.
840///
841/// # Arguments
842/// - `read` - The read to check.
843/// - `kmer_size` - The size of the k-mers to search for.
844/// - `min_kmers_pct` - The minimum percentage of k-mers to search for.
845/// - `kmer_set` - The set of k-mers to search for.
846///
847/// # Returns
848/// Whether the read is valid or not.
849fn is_valid_read(
850    read: &BamRecord,
851    kmer_size: usize,
852    min_kmers_pct: usize,
853    kmer_set: &HashSet<Vec<u8>>,
854) -> bool {
855    let read_seq = read.seq().as_bytes();
856    let rl_seq = skydive::utils::homopolymer_compressed(&read_seq);
857    let num_kmers = rl_seq
858        .par_windows(kmer_size)
859        .filter(|kmer| kmer_set.contains(&skydive::utils::canonicalize_kmer(kmer)))
860        .count();
861    (num_kmers as f32 / rl_seq.len() as f32) > (min_kmers_pct as f32 / 100.0)
862}

864/// This function finalizes the progress.
865///
866/// # Arguments
867/// - `progress_bar` - The progress bar to finalize.
868/// - `found_items` - The number of found items.
869/// - `processed_items` - The number of processed items.
870fn finalize_progress(
871    progress_bar: &Arc<ProgressBar>,
872    found_items: &Arc<AtomicUsize>,
873    processed_items: &Arc<AtomicUsize>,
874) {
875    let final_found = found_items.load(Ordering::Relaxed);
876    let final_processed = processed_items.load(Ordering::Relaxed);
877    progress_bar.set_message(format!(
878        "Found {} reads in {} reads total.",
879        final_found.to_formatted_string(&Locale::en),
880        final_processed.to_formatted_string(&Locale::en)
881    ));
882    progress_bar.finish();
883}
