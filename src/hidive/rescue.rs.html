37pub fn start(
38    output: &PathBuf,
39    kmer_size: usize,
40    min_kmers_pct: usize,
41    search_option: SearchOption,
42    ref_path: Option<PathBuf>,
43    loci: Option<Vec<String>>,
44    fastx_paths: &Vec<PathBuf>,
45    seq_paths: &Vec<PathBuf>,
46) {
47    let fastx_urls = skydive::parse::parse_file_names(fastx_paths);
48    let seq_urls = skydive::parse::parse_file_names(seq_paths);
49
50    // Get the system's temporary directory path
51    let cache_path = std::env::temp_dir();
52    skydive::elog!("Intermediate data will be stored at {:?}.", cache_path);
53
54    // Define the fetch definitions for the aligner.
55    let mut fetches = Vec::new();
56
57    // Read the FASTA/FASTQ files and prepare a hashmap of k-mers to search for in the other files.
58    let mut kmer_set = HashSet::new();
59    for fastx_url in fastx_urls {
60        let file_path = fastx_url.to_file_path().unwrap();
61        let mut reads = Vec::new();
62        
63        // Use needletail's parse_fastx_file which automatically detects FASTA/FASTQ format
64        let mut reader = parse_fastx_file(&file_path).expect("Failed to open sequence file");
65        
66        while let Some(record) = reader.next() {
67            let record = record.expect("Failed to parse sequence record");
68            let fw_seq = record.seq();
69            let rl_seq = skydive::utils::homopolymer_compressed(&fw_seq);
70
71            let kmers = rl_seq
72                .windows(kmer_size)
73                .map(skydive::utils::canonicalize_kmer)
74                .collect::<HashSet<Vec<u8>>>();
75
76            kmer_set.extend(kmers);
77            reads.push(fw_seq.to_vec());
78        }
79
80        skydive::elog!("Read {} reads from {}.", reads.len(), file_path.to_str().unwrap());
81
82        // Set fetches with the input loci or the detected relevant loci.
83        if search_option == SearchOption::Contig || search_option == SearchOption::ContigAndInterval {
84            match populate_fetches(&search_option, &loci, &ref_path, &reads, &mut fetches) {
85                Ok(_) => {}
86                Err(e) => {
87                    skydive::elog!("Error: {}", e);
88                    std::process::exit(1);
89                }
90            }
91            print_fetches_info(&fetches);
92        } else {
93            assert!(
94                (search_option == SearchOption::All || search_option == SearchOption::Unmapped)
95                    && loci.is_none(),
96                "Assertion failed: search_option is 'All' or 'Unmapped' and loci is NOT None"
97            );
98        }
99    }
100
101    skydive::elog!("Fetches: {:?}", fetches);
102
103    // Read the CRAM files and search for the k-mers in each read.
104    let mut all_records = Vec::new();
105    for seq_url in seq_urls {
106        let mut reader = skydive::stage::open_bam(&seq_url).unwrap();
107
108        let progress_bar =
109            skydive::utils::default_unbounded_progress_bar("Searching for similar reads (0 found)");
110
111        // Create some thread-safe counters.
112        let found_items = Arc::new(AtomicUsize::new(0));
113        let processed_items = Arc::new(AtomicUsize::new(0));
114        let progress_bar = Arc::new(progress_bar);
115
116        // const UPDATE_FREQUENCY: usize = 1_000_000;
117
118        // Create an immutable map of tid to chromosome name
119        let tid_to_chrom: HashMap<i32, String> = reader
120            .header()
121            .target_names()
122            .iter()
123            .enumerate()
124            .map(|(tid, &name)| (tid as i32, String::from_utf8_lossy(name).into_owned()))
125            .collect::<HashMap<_, _>>();
126
127        if search_option == SearchOption::Contig || search_option == SearchOption::ContigAndInterval
128        {
129            // Use the fetches to retrieve the records.
130            retrieve_records(
131                &mut reader,
132                Some(&fetches),
133                &search_option,
134                kmer_size,
135                min_kmers_pct,
136                &kmer_set,
137                &tid_to_chrom,
138                &progress_bar,
139                &found_items,
140                &processed_items,
141                &mut all_records,
142            );
143        } else {
144            // Retrieve all or unmapped reads.
145            retrieve_records(
146                &mut reader,
147                None,
148                &search_option,
149                kmer_size,
150                min_kmers_pct,
151                &kmer_set,
152                &tid_to_chrom,
153                &progress_bar,
154                &found_items,
155                &processed_items,
156                &mut all_records,
157            );
158        }
159
160        let buf_writer = BufWriter::new(File::create(output).unwrap());
161        let mut fastq_writer = fastq::Writer::new(buf_writer);
162
163        write_paired_fastq_records(&all_records, &mut fastq_writer);
164
165        if !all_records.is_empty() {
166            skydive::elog!(
167                "Wrote {} reads to {}.",
168                all_records.len().to_formatted_string(&Locale::en),
169                output.to_str().unwrap()
170            );
171        } else {
172            skydive::elog!("No reads were found in the CRAM files. Aborting.");
173            std::process::exit(1);
174        }
175    }
176}

183/// This function will populate the fetches with the input loci or the detected relevant loci.
184/// Function should only be called if the search option is set to 'Contig' or 'ContigAndInterval'.
185///
186/// # Arguments
187/// - `search_option` - The search option.
188/// - `loci` - The loci to search for.
189/// - `ref_path` - The path to the reference genome.
190/// - `reads` - The reads to map.
191/// - `fetches` - The fetches to populate.
192///
193/// # Panics
194/// If the search option is set to 'Contig' or 'ContigAndInterval' and no reference genome is provided.
195fn populate_fetches(
196    search_option: &SearchOption,
197    loci: &Option<Vec<String>>,
198    ref_path: &Option<PathBuf>,
199    reads: &Vec<Vec<u8>>,
200    fetches: &mut Vec<(String, Interval<i32>)>,
201) -> Result<(), String> {
202    if !matches!(
203        search_option,
204        SearchOption::Contig | SearchOption::ContigAndInterval
205    ) {
206        return Err("Function should only be called if the search option is 'Contig' or 'ContigAndInterval'".to_string());
207    }
208
209    if let Some(loci) = loci {
210        skydive::elog!("Setting loci from input...");
211        let loci = skydive::parse::parse_loci(loci, 0);
212        for (contig, start, stop, _) in loci {
213            match Interval::new(start as i32..stop as i32) {
214                Ok(interval) => fetches.push((contig, interval)),
215                Err(e) => {
216                    skydive::elog!(
217                        "Invalid interval for contig {}: {:?}, start: {}, stop: {}. Error: {}",
218                        contig,
219                        start..stop,
220                        start,
221                        stop,
222                        e
223                    );
224                    return Err(format!(
225                        "Error creating interval for contig {}: {:?}",
226                        contig, e
227                    ));
228                }
229            }
230        }
231    } else if let Some(ref ref_path) = ref_path {
232        skydive::elog!("Searching for relevant loci in reference genome...");
233        detect_relevant_loci(ref_path, reads, fetches);
234        fetches.push(("*".to_string(), Interval::new(0..1).unwrap()));
235    } else {
236        skydive::elog!("No reference genome provided but search option is set to 'Contig' or 'ContigAndInterval'. Aborting.");
237        return Err("No reference genome provided.".to_string());
238    }
239
240    Ok(())
241}

243/// This function will print the fetches' info.
244/// # Arguments
245/// - `fetches` - The fetches to print.
246///
247/// # Panics
248///
249/// If the total length of the fetches cannot be calculated.
250fn print_fetches_info(fetches: &[(String, Interval<i32>)]) {
251    let total_length = fetches
252        .iter()
253        .map(|(_, interval)| interval.end - interval.start)
254        .sum::<i32>();
255    skydive::elog!(
256        " -- will search unaligned reads and {} bases in {} contigs.",
257        total_length.to_formatted_string(&Locale::en),
258        fetches.len().to_formatted_string(&Locale::en)
259    );
260
261    for (contig, interval) in fetches {
262        skydive::elog!(" -- {}:{:?}", contig, interval);
263    }
264}

271/// This function detects the relevant loci in the reference genome.
272///
273/// # Arguments
274///
275/// * `ref_path` - The path to the reference genome.
276/// * `reads` - The reads to map.
277/// * `fetches` - The fetches to populate.
278///
279/// # Panics
280///
281/// If no mappings are found for the provided sequences.
282fn detect_relevant_loci(
283    ref_path: &PathBuf,
284    reads: &Vec<Vec<u8>>,
285    fetches: &mut Vec<(String, Interval<i32>)>,
286) {
287    let lr_aligner = initialize_aligner(ref_path, false);
288
289    let lr_mappings = map_sequences(&lr_aligner, &reads, false);
290
291    for lr_mapping in &lr_mappings {
292        skydive::elog!(
293            "lr mapping {:?}:{}-{}",
294            lr_mapping.target_name,
295            lr_mapping.target_start,
296            lr_mapping.target_end
297        );
298    }
299
300    let sr_aligner = initialize_aligner(ref_path, true);
301
302    let sr_mappings = map_sequences(&sr_aligner, &reads, true);
303
304    let mut contig_lengths = HashMap::new();
305    let loci = collect_loci(&lr_mappings, &sr_mappings, &mut contig_lengths);
306
307    // First pass - put all the intervals into a tree.
308    let mut trees = BTreeMap::new();
309    populate_interval_trees(&mut trees, &loci, &contig_lengths);
310
311    // Second pass
312    merge_overlapping_intervals(&mut trees, &loci, &contig_lengths);
313
314    let sub_fetches: HashSet<_> = trees
315        .iter()
316        .flat_map(|(contig_name, tree)| {
317            tree.find(Interval::new(0..i32::MAX).unwrap())
318                .map(move |entry| (contig_name.clone(), entry.interval().clone()))
319        })
320        .collect();
321
322    fetches.extend(sub_fetches.into_iter());
323}

325/// This function initializes the aligner.
326///
327/// # Arguments
328///
329/// * `ref_path` - The path to the reference genome.
330/// * `is_sr` - Whether the reads are short reads or not.
331///
332/// # Returns
333///
334/// An aligner object.
335///
336/// # Panics
337///
338/// If the index cannot be built.
339fn initialize_aligner(ref_path: &PathBuf, is_sr: bool) -> Aligner<Built> {
340    if is_sr {
341        Aligner::builder()
342            .sr()
343            .with_sam_hit_only()
344            .with_index(ref_path, None)
345            .expect("Unable to build index")
346    } else {
347        Aligner::builder()
348            .map_hifi()
349            .with_sam_hit_only()
350            .with_index(ref_path, None)
351            .expect("Unable to build index")
352    }
353}

355/// This function maps the sequences to the reference genome using the aligner.
356///
357/// # Arguments
358///
359/// * `aligner` - The aligner object to use for mapping.
360/// * `reads` - The reads to map.
361/// * `is_sr` - Whether the reads are short reads or not.
362///
363/// # Returns
364///
365/// A vector of mappings.
366///
367/// # Panics
368///
369/// If no mappings are found for the provided sequences.
370fn map_sequences(aligner: &Aligner<Built>, reads: &[Vec<u8>], is_sr: bool) -> Vec<Mapping> {
371    let mappings: Vec<Mapping> = reads
372        .par_iter()
373        .flat_map(|seq| {
374            if is_sr {
375                seq.windows(150)
376                    .flat_map(|window| {
377                        aligner
378                            .map(window, false, false, None, None, None)
379                            .unwrap_or_else(|_| vec![])
380                    })
381                    .collect::<Vec<_>>()
382            } else {
383                aligner
384                    .map(seq, false, false, None, None, None)
385                    .unwrap_or_else(|_| vec![])
386            }
387        })
388        .collect();
389
390    if mappings.is_empty() {
391        panic!("No mappings found for the provided sequences.");
392    }
393
394    mappings
395}

397/// This function collects the loci from the mappings.
398///
399/// # Arguments
400///
401/// * `lr_mappings` - The long read mappings.
402/// * `sr_mappings` - The short read mappings.
403///
404/// # Returns
405///
406/// A hashset of loci.
407///
408/// # Panics
409///
410/// If the target name is not found in the mappings.
411fn collect_loci(
412    lr_mappings: &[Mapping],
413    sr_mappings: &[Mapping],
414    contig_lengths: &mut HashMap<String, i32>,
415) -> HashSet<(Vec<u8>, i32, i32)> {
416    lr_mappings
417        .iter()
418        .chain(sr_mappings.iter())
419        .filter_map(|m| {
420            if let Some(target_name) = &m.target_name {
421                contig_lengths.insert(target_name.to_string(), m.target_len);
422                Some((
423                    target_name.to_string().as_bytes().to_vec(),
424                    m.target_start,
425                    m.target_end,
426                ))
427            } else {
428                None
429            }
430        })
431        .collect::<HashSet<_>>()
432}

434/// This function populates the interval trees with the loci.
435///
436/// # Arguments
437///
438/// * `trees` - The interval trees to populate.
439/// * `loci` - The loci to insert into the trees.
440/// * `contig_lengths` - The lengths of the contigs.
441///
442/// # Panics
443///
444/// If the contig name is not found in the contig lengths.
445fn populate_interval_trees(
446    trees: &mut BTreeMap<String, IntervalTree<i32, ()>>,
447    loci: &HashSet<(Vec<u8>, i32, i32)>,
448    contig_lengths: &HashMap<String, i32>,
449) {
450    for (seq, start, end) in loci {
451        let contig_name = String::from_utf8_lossy(&seq).to_string();
452        let contig_length = contig_lengths.get(&contig_name).unwrap();
453
454        if !trees.contains_key(&contig_name) {
455            trees.insert(contig_name.clone(), IntervalTree::new());
456        }
457
458        let interval = Interval::new(
459            start.saturating_sub(50000).max(0)..end.saturating_add(50000).min(*contig_length),
460        )
461        .unwrap();
462        trees.get_mut(&contig_name).unwrap().insert(interval, ());
463    }
464}

466/// This function merges overlapping intervals.
467///
468/// # Arguments
469///
470/// * `trees` - The interval trees to merge.
471/// * `loci` - The loci to merge.
472/// * `contig_lengths` - The lengths of the contigs.
473///
474/// # Panics
475///
476/// If the contig name is not found in the trees.
477fn merge_overlapping_intervals(
478    trees: &mut BTreeMap<String, IntervalTree<i32, ()>>,
479    loci: &HashSet<(Vec<u8>, i32, i32)>,
480    contig_lengths: &HashMap<String, i32>,
481) {
482    for (seq, start, end) in loci {
483        let contig_name = String::from_utf8_lossy(&seq).to_string();
484        let contig_length = contig_lengths.get(&contig_name).unwrap();
485        let tree = trees.get_mut(&contig_name).unwrap();
486
487        let current_interval = Interval::new(
488            start.saturating_sub(50000).max(0)..end.saturating_add(50000).min(*contig_length),
489        )
490        .unwrap();
491        let overlaps = tree.find(current_interval.clone()).collect::<Vec<_>>();
492
493        let mut merged_intervals = IntervalTree::new();
494
495        if !overlaps.is_empty() {
496            let min_start = overlaps
497                .iter()
498                .map(|o| o.interval().start)
499                .min()
500                .unwrap()
501                .min(*start);
502            let max_end = overlaps
503                .iter()
504                .map(|o| o.interval().end)
505                .max()
506                .unwrap()
507                .max(*end);
508            merged_intervals.insert(Interval::new(min_start..max_end).unwrap(), ());
509        } else {
510            merged_intervals.insert(current_interval, ());
511        }
512
513        trees.insert(contig_name.clone(), merged_intervals);
514    }
515}

522/// This function retrieves the records from the reader.
523///
524/// # Arguments
525/// - `fetches` - The fetches to process.
526/// - `reader` - The reader to use for fetching reads.
527/// - `kmer_size` - The size of the k-mers to search for.
528/// - `min_kmers_pct` - The minimum percentage of k-mers to search for.
529/// - `search_option` - Whether to search all reads or not.
530/// - `kmer_set` - The set of k-mers to search for.
531/// - `tid_to_chrom` - The mapping of TID to chromosome name.
532/// - `progress_bar` - The progress bar to update.
533/// - `found_items` - The number of found items.
534/// - `processed_items` - The number of processed items.
535/// - `all_records` - The vector of all records.
536///
537fn retrieve_records(
538    reader: &mut IndexedReader,
539    fetches: Option<&[(String, Interval<i32>)]>,
540    search_option: &SearchOption,
541    kmer_size: usize,
542    min_kmers_pct: usize,
543    kmer_set: &HashSet<Vec<u8>>,
544    tid_to_chrom: &HashMap<i32, String>,
545    progress_bar: &Arc<ProgressBar>,
546    found_items: &Arc<AtomicUsize>,
547    processed_items: &Arc<AtomicUsize>,
548    all_records: &mut Vec<BamRecord>,
549) {
550    // const UPDATE_FREQUENCY: usize = 1_000_000;
551
552    if let Some(fetches) = fetches {
553        let fetches_updated = prepare_fetches(fetches, search_option);
554
555        for (contig, interval) in fetches_updated {
556            let fetch_definition =
557                create_fetch_definition(search_option, Some(&contig), Some(interval));
558            reader
559                .fetch(fetch_definition)
560                .expect("Failed to fetch reads");
561
562            let records = filter_and_collect_records(
563                reader,
564                kmer_size,
565                min_kmers_pct,
566                kmer_set,
567                tid_to_chrom,
568                progress_bar,
569                found_items,
570                processed_items,
571            );
572
573            all_records.extend(records);
574        }
575    } else {
576        let fetch_definition = create_fetch_definition(search_option, None, None);
577        reader
578            .fetch(fetch_definition)
579            .expect("Failed to fetch reads");
580
581        let records = filter_and_collect_records(
582            reader,
583            kmer_size,
584            min_kmers_pct,
585            kmer_set,
586            tid_to_chrom,
587            progress_bar,
588            found_items,
589            processed_items,
590        );
591
592        all_records.extend(records);
593    }
594
595    finalize_progress(progress_bar, found_items, processed_items);
596}

598/// This function processes all or unmapped reads.
599/// # Arguments
600/// - `reader` - The reader to use for fetching reads.
601/// - `kmer_size` - The size of the k-mers to search for.
602/// - `min_kmers_pct` - The minimum percentage of k-mers to search for.
603/// - `kmer_set` - The set of k-mers to search for.
604/// - `tid_to_chrom` - The mapping of TID to chromosome name.
605/// - `progress_bar` - The progress bar to update.
606/// - `found_items` - The number of found items.
607/// - `processed_items` - The number of processed items.
608///
609/// # Panics
610/// If the search option is not 'All' or 'Unmapped'.
611///
612/// # Returns
613/// A vector of all records.
614fn filter_and_collect_records(
615    reader: &mut IndexedReader,
616    kmer_size: usize,
617    min_kmers_pct: usize,
618    kmer_set: &HashSet<Vec<u8>>,
619    tid_to_chrom: &HashMap<i32, String>,
620    progress_bar: &Arc<ProgressBar>,
621    found_items: &Arc<AtomicUsize>,
622    processed_items: &Arc<AtomicUsize>,
623) -> Vec<BamRecord> {
624    const UPDATE_FREQUENCY: usize = 1_000_000;
625
626    reader
627        .records()
628        .par_bridge()
629        .flat_map(|record| record.ok())
630        .filter_map(|read| {
631            update_processed_progress(
632                processed_items,
633                UPDATE_FREQUENCY,
634                tid_to_chrom,
635                &read,
636                &progress_bar,
637            );
638
639            if is_valid_read(&read, kmer_size, min_kmers_pct, kmer_set) {
640                update_found_progress(
641                    found_items,
642                    UPDATE_FREQUENCY,
643                    tid_to_chrom,
644                    &read,
645                    &progress_bar,
646                );
647                Some(read)
648            } else {
649                None
650            }
651        })
652        .collect()
653}

655/// This function prepares the fetches for searching.
656/// If the search option is 'Contig', it will extract the unique contigs from the fetches and add a dummy interval.
657/// If the search option is 'ContigAndInterval', it will return the fetches as is.
658/// In both cases, the fetches will be returned as a vector of contigs and intervals.
659///
660/// # Arguments
661/// - `fetches` - The fetches to prepare.
662/// - `search_option` - Whether to search all reads or not.
663///
664/// # Returns
665/// A vector of fetches to search.
666///
667/// # Panics
668/// If the contig is not valid.
669fn prepare_fetches(
670    fetches: &[(String, Interval<i32>)],
671    search_option: &SearchOption,
672) -> Vec<(String, Interval<i32>)> {
673    if *search_option == SearchOption::Contig {
674        let uniq_contigs: HashSet<_> = fetches.iter().map(|(contig, _)| contig.clone()).collect();
675        uniq_contigs
676            .into_iter()
677            .map(|contig| (contig, Interval::new(0..1).unwrap()))
678            .collect()
679    } else if *search_option == SearchOption::ContigAndInterval {
680        fetches
681            .iter()
682            .map(|(contig, interval)| (contig.clone(), interval.clone()))
683            .collect()
684    } else {
685        panic!("Invalid search option");
686    }
687}

689/// This function creates a fetch definition.
690///
691/// # Arguments
692/// - `search_option` - Whether to search all reads or not.
693/// - `contig` - The contig to fetch.
694/// - `interval` - The interval to fetch.
695///
696/// # Returns
697/// A fetch definition.
698fn create_fetch_definition<'a>(
699    search_option: &'a SearchOption,
700    contig: Option<&'a String>,
701    interval: Option<Interval<i32>>,
702) -> FetchDefinition<'a> {
703    match (contig, search_option) {
704        (None, SearchOption::All) => FetchDefinition::All,
705        (None, SearchOption::Unmapped) => FetchDefinition::Unmapped,
706        (Some(contig), SearchOption::Contig) if contig != "*" => {
707            FetchDefinition::String(contig.as_bytes())
708        }
709        (Some(contig), SearchOption::ContigAndInterval) if contig != "*" => {
710            let interval = interval.unwrap_or_else(|| Interval::new(0..1).unwrap());
711            FetchDefinition::RegionString(
712                contig.as_bytes(),
713                interval.start as i64,
714                interval.end as i64,
715            )
716        }
717        (Some(contig), _) if contig == "*" => FetchDefinition::Unmapped,
718        _ => panic!(
719            "Invalid fetch definition settings. \
720                If you are using the 'Contig' or 'ContigAndInterval' search options, \
721                you must provide a contig name. \
722                If you are using the 'All' or 'Unmapped' search options, \
723                you must NOT provide a contig name. \
724             search option: '{:?}' , contig: '{:?}', interval: '{:?}'",
725            search_option,
726            contig.as_deref().unwrap_or(&"None".to_string()),
727            interval
728                .map(|i| format!("{:?}", i))
729                .unwrap_or_else(|| "None".to_string())
730        ),
731    }
732}

734/// This function updates the processed progress.
735///
736/// # Arguments
737/// - `processed_items` - The number of processed items.
738/// - `update_freq` - The update frequency.
739/// - `tid_to_chrom` - The mapping of TID to chromosome name.
740/// - `read` - The read to process.
741/// - `progress_bar` - The progress bar to update.
742///
743/// # Panics
744/// If the chromosome name is not found in the mapping.
745fn update_processed_progress(
746    processed_items: &Arc<AtomicUsize>,
747    update_freq: usize,
748    tid_to_chrom: &HashMap<i32, String>,
749    read: &BamRecord,
750    progress_bar: &Arc<ProgressBar>,
751) {
752    let current_processed = processed_items.fetch_add(1, Ordering::Relaxed);
753    if current_processed % update_freq == 0 {
754        let unknown_chrom = "Unknown".to_string();
755        let chrom_name = tid_to_chrom.get(&read.tid()).unwrap_or(&unknown_chrom);
756        progress_bar.set_message(format!(
757            "Searching for similar reads ({} processed, most recent at {}:{})",
758            current_processed.to_formatted_string(&Locale::en),
759            chrom_name,
760            read.reference_start().to_formatted_string(&Locale::en)
761        ));
762        progress_bar.inc(update_freq as u64);
763    }
764}

797/// This function checks if a read is valid.
798///
799/// # Arguments
800/// - `read` - The read to check.
801/// - `kmer_size` - The size of the k-mers to search for.
802/// - `min_kmers_pct` - The minimum percentage of k-mers to search for.
803/// - `kmer_set` - The set of k-mers to search for.
804///
805/// # Returns
806/// Whether the read is valid or not.
807fn is_valid_read(
808    read: &BamRecord,
809    kmer_size: usize,
810    min_kmers_pct: usize,
811    kmer_set: &HashSet<Vec<u8>>,
812) -> bool {
813    let read_seq = read.seq().as_bytes();
814    let rl_seq = skydive::utils::homopolymer_compressed(&read_seq);
815    let num_kmers = rl_seq
816        .par_windows(kmer_size)
817        .filter(|kmer| kmer_set.contains(&skydive::utils::canonicalize_kmer(kmer)))
818        .count();
819    (num_kmers as f32 / rl_seq.len() as f32) > (min_kmers_pct as f32 / 100.0)
820}

822/// This function finalizes the progress.
823///
824/// # Arguments
825/// - `progress_bar` - The progress bar to finalize.
826/// - `found_items` - The number of found items.
827/// - `processed_items` - The number of processed items.
828fn finalize_progress(
829    progress_bar: &Arc<ProgressBar>,
830    found_items: &Arc<AtomicUsize>,
831    processed_items: &Arc<AtomicUsize>,
832) {
833    let final_found = found_items.load(Ordering::Relaxed);
834    let final_processed = processed_items.load(Ordering::Relaxed);
835    progress_bar.set_message(format!(
836        "Found {} reads in {} reads total.",
837        final_found.to_formatted_string(&Locale::en),
838        final_processed.to_formatted_string(&Locale::en)
839    ));
840    progress_bar.finish();
841}

843fn write_paired_fastq_records(records: &[BamRecord], fastq_writer: &mut fastq::Writer<BufWriter<File>>) {
844    let mut grouped_records = HashMap::new();
845    let mut emitted_names = HashSet::new();
846    
847    for record in records {
848        let qname = String::from_utf8_lossy(record.qname()).into_owned();
849        grouped_records.entry(qname).or_insert_with(Vec::new).push(record);
850    }
851
852    for (qname, records) in grouped_records {
853        // Skip if we've already emitted this read name
854        if emitted_names.contains(&qname) {
855            continue;
856        }
857        
858        // Handle proper paired reads (exactly 2 records)
859        if records.len() == 2 {
860            let mut paired_records = Vec::new();
861            for record in records {
862                if record.is_reverse() {
863                    let rv_seq = record.seq().as_bytes().reverse_complement();
864                    let mut rv_qual = record.qual().iter().map(|&q| q + 33).collect::<Vec<u8>>();
865                    rv_qual.reverse();
866                    let rv_record = fastq::Record::with_attrs(&String::from_utf8_lossy(record.qname()), None, &rv_seq, &rv_qual);
867                    paired_records.push(rv_record);
868                } else {
869                    let fw_seq = record.seq().as_bytes();
870                    let fw_qual = record.qual().iter().map(|&q| q + 33).collect::<Vec<u8>>();
871                    let fw_record = fastq::Record::with_attrs(&String::from_utf8_lossy(record.qname()), None, &fw_seq, &fw_qual);
872                    paired_records.push(fw_record);
873                }
874            }
875            fastq_writer.write_record(&paired_records[0]).unwrap();
876            fastq_writer.write_record(&paired_records[1]).unwrap();
877            
878            // Mark this read name as emitted
879            emitted_names.insert(qname);
880        }
881    }
882}
