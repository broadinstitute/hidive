1use num_format::{Locale, ToFormattedString};
2use std::path::PathBuf;
3
4
5use crate::train::{create_dataset_for_model, distance_to_a_contig_end, plot_roc_curve, process_reads};
6use gbdt::decision_tree::DataVec;
7use gbdt::gradient_boost::GBDT;
8use skydive::ldbg::LdBG;
9use std::io::Write;
10
11pub fn start(
12    output: &PathBuf,
13    kmer_size: usize,
14    long_read_seq_paths: &Vec<PathBuf>,
15    short_read_seq_paths: &Vec<PathBuf>,
16    truth_seq_paths: &Vec<PathBuf>,
17    model_path: &PathBuf,
18) {
19    let long_read_seq_urls = skydive::parse::parse_file_names(long_read_seq_paths);
20    let short_read_seq_urls = skydive::parse::parse_file_names(short_read_seq_paths);
21    let truth_seq_urls = skydive::parse::parse_file_names(truth_seq_paths);
22
23
24    // Read all long reads.
25    let all_lr_seqs: Vec<Vec<u8>> = process_reads(&long_read_seq_urls, "long");
26    let l1 = LdBG::from_sequences(String::from("l1"), kmer_size, &all_lr_seqs);
27
28    // Read all short reads.
29    let all_sr_seqs: Vec<Vec<u8>> = process_reads(&short_read_seq_urls, "short");
30    let s1 = LdBG::from_sequences(String::from("s1"), kmer_size, &all_sr_seqs);
31
32    // Read all truth sequences.
33    let all_truth_seqs: Vec<Vec<u8>> = process_reads(&truth_seq_urls, "truth");
34    let t1 = LdBG::from_sequences(String::from("s1"), kmer_size, &all_truth_seqs);
35
36    let lr_contigs = l1.assemble_all();
37    let lr_distances = distance_to_a_contig_end(&lr_contigs, kmer_size);
38
39    let sr_contigs = s1.assemble_all();
40    let sr_distances = distance_to_a_contig_end(&sr_contigs, kmer_size);
41
42
43    // load model
44    skydive::elog!("Loading GBDT model from {}...", model_path.to_str().unwrap());
45    let gbdt = GBDT::load_model(model_path.to_str().unwrap()).expect("Unable to load model");
46
47    // Prepare test data.
48    let test_kmers = l1
49        .kmers
50        .keys()
51        .chain(s1.kmers.keys())
52        .chain(t1.kmers.keys());
53    let test_data: DataVec = create_dataset_for_model(
54        test_kmers,
55        &lr_distances,
56        &sr_distances,
57        &l1,
58        &s1,
59        &t1,
60    );
61
62    // Predict the test data.
63    skydive::elog!("Computing accuracy on test data...");
64    let prediction = gbdt.predict(&test_data);
65    let pred_threshold = 0.5;
66
67    // Evaluate accuracy of the model on the test data.
68    let mut num_correct = 0u32;
69    let mut num_total = 0u32;
70    for (data, pred) in test_data.iter().zip(prediction.iter()) {
71        let truth = data.label;
72        let call = if *pred > pred_threshold { 1.0 } else { 0.0 };
73        if (call - truth).abs() < f32::EPSILON {
74            num_correct += 1;
75        }
76        num_total += 1;
77    }
78
79    // Precision, Recall, and F1 score calculations.
80    let (precision, recall, f1_score) = crate::train::compute_precision_recall_f1(&test_data, &prediction, pred_threshold);
81    skydive::elog!("Prediction threshold: {:.2}", pred_threshold);
82    skydive::elog!("Precision: {:.2}%", 100.0 * precision);
83    skydive::elog!("Recall: {:.2}%", 100.0 * recall);
84    skydive::elog!("F1 score: {:.2}%", 100.0 * f1_score);
85
86    // TPR and FPR calculations at various thresholds.
87    let fpr_tpr = crate::train::compute_fpr_tpr(&test_data, &prediction);
88
89    // Save TPR and FPR at various thresholds to a file.
90    let csv_output = output.with_extension("csv");
91    let mut writer = std::fs::File::create(&csv_output).expect("Unable to create file");
92    for (tpr, fpr) in &fpr_tpr {
93        writeln!(writer, "{},{}", tpr, fpr).expect("Unable to write data");
94    }
95    skydive::elog!("TPR and FPR at various thresholds saved to {}", csv_output.to_str().unwrap());
96
97    // Create a ROC curve.
98    let png_output = output.with_extension("png");
99    plot_roc_curve(&png_output, &fpr_tpr).expect("Unable to plot ROC curve");
100    skydive::elog!("ROC curve saved to {}", png_output.to_str().unwrap());
101
102    skydive::elog!(
103        "Predicted accuracy: {}/{} ({:.2}%)",
104        num_correct.to_formatted_string(&Locale::en),
105        num_total.to_formatted_string(&Locale::en),
106        100.0 * num_correct as f32 / num_total as f32
107    );
108
109    gbdt.save_model(output.to_str().unwrap())
110        .expect("Unable to save model");
111
112    skydive::elog!("Model saved to {}", output.to_str().unwrap());
113}
