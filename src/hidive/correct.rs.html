13pub fn start(
14    output: &PathBuf,
15    loci_list: &Vec<String>,
16    kmer_size: usize,
17    window: usize,
18    model_path: &PathBuf,
19    long_read_fasta_path: &PathBuf,
20    short_read_fasta_path: &PathBuf,
21) {
22    // Get the system's temporary directory path
23    let cache_path = std::env::temp_dir();
24    skydive::elog!("Intermediate data will be stored at {:?}.", cache_path);
25
26    let model_absolute_path = if model_path.is_absolute() {
27        model_path.clone()
28    } else {
29        std::env::current_dir().unwrap().join(model_path)
30    };
31
32    // Load datasets
33    let long_read_seq_urls = skydive::parse::parse_file_names(&[long_read_fasta_path.clone()]);
34
35    skydive::elog!(
36        "Processing short-read sample {}...",
37        short_read_fasta_path.display()
38    );
39    let all_sr_seqs = skydive::utils::read_fasta(&vec![short_read_fasta_path.clone()]);
40    skydive::elog!(" - {} short reads loaded", all_sr_seqs.len());
41
42    // Iterate over loci
43    let loci = skydive::parse::parse_loci(loci_list, window as u64)
44        .into_iter()
45        .collect::<Vec<_>>();
46
47    // let progress_bar = skydive::utils::default_bounded_progress_bar("Processing loci", loci.len() as u64);
48
49    let fa_file = std::sync::Mutex::new(File::create(output).unwrap());
50    loci
51        // .par_iter()
52        // .progress_with(progress_bar)
53        .iter()
54        .inspect(|(chrom, start, end, name)| {
55            skydive::elog!("Processing locus {} ({}:{}-{})", name, chrom, start, end);
56        })
57        .for_each(|(chrom, start, end, name)| {
58            let (padded_start, padded_end) = pad_interval(start, end, window);
59
60            // let mut corrected_reads_old: HashMap<String, Vec<(Vec<u8>, HashMap<Vec<u8>, f32>)>> = HashMap::new();
61            // for window_start in (padded_start..padded_end).step_by(window) {
62
63            let corrected_reads: HashMap<String, Vec<(Vec<u8>, HashMap<Vec<u8>, f32>)>> =
64                (padded_start..padded_end)
65                    .step_by(window)
66                    .collect::<Vec<_>>()
67                    .into_par_iter()
68                    .filter_map(|window_start| {
69                        let window_end = window_start + window as u64;
70
71                        let mut locus = LinkedHashSet::new();
72                        locus.insert((chrom.clone(), window_start, window_end, name.clone()));
73
74                        let r = skydive::stage::stage_data_in_memory(
75                            &locus,
76                            &long_read_seq_urls,
77                            false,
78                            &cache_path,
79                        );
80
81                        if let Ok(reads) = r {
82                            Some(reads)
83                        } else {
84                            None
85                        }
86                    })
87                    .map(|reads| {
88                        let long_reads: HashMap<String, Vec<u8>> = reads
89                            .into_iter()
90                            .map(|read| (read.id().to_string(), read.seq().to_vec()))
91                            .collect();
92
93                        let cn_kmers = long_reads
94                            .values()
95                            .map(|seq| seq.windows(kmer_size).collect::<Vec<_>>())
96                            .flatten()
97                            .map(|kmer| skydive::utils::canonicalize_kmer(kmer))
98                            .collect::<HashSet<_>>();
99
100                        let sr_seqs = all_sr_seqs
101                            .iter()
102                            .filter_map(|seq| {
103                                let kmers = seq.windows(kmer_size).collect::<Vec<_>>();
104                                let read_kmers = kmers
105                                    .iter()
106                                    .map(|kmer| skydive::utils::canonicalize_kmer(kmer))
107                                    .collect::<HashSet<_>>();
108
109                                let count = cn_kmers.intersection(&read_kmers).count();
110
111                                if count > (0.5 * read_kmers.len() as f64) as usize {
112                                    Some(seq)
113                                } else {
114                                    None
115                                }
116                            })
117                            .cloned()
118                            .collect::<Vec<_>>();
119
120                        let lr_seqs = long_reads.values().cloned().collect::<Vec<_>>();
121                        let l1 = LdBG::from_sequences("lr".to_string(), kmer_size, &lr_seqs);
122                        let s1 = LdBG::from_sequences("sr".to_string(), kmer_size, &sr_seqs);
123
124                        let m = MLdBG::from_ldbgs(vec![l1, s1])
125                            .score_kmers(&model_absolute_path)
126                            .collapse()
127                            .clean(0.1)
128                            .build_links(&lr_seqs, false);
129
130                        let g = m.traverse_all_kmers();
131
132                        let mut window_corrections = HashMap::new();
133                        for (id, seq) in long_reads {
134                            let corrected_seq = m.correct_seq(&g, &seq);
135                            window_corrections
136                                .entry(id)
137                                .or_insert_with(Vec::new)
138                                .push((corrected_seq, m.scores.clone()));
139                        }
140
141                        window_corrections
142                    })
143                    .reduce(
144                        || HashMap::new(),
145                        |mut acc, window_map| {
146                            for (id, corrections) in window_map {
147                                acc.entry(id).or_insert_with(Vec::new).extend(corrections);
148                            }
149                            acc
150                        },
151                    );
152
153            let mut file = fa_file.lock().unwrap();
154            for id in corrected_reads.keys() {
155                let pieces = corrected_reads.get(id).unwrap();
156
157                let mut joined_seq = Vec::<u8>::new();
158                let mut joined_scores = HashMap::new();
159                for (piece, scores) in pieces {
160                    joined_seq.extend(piece);
161                    joined_scores.extend(scores);
162                }
163
164                let mut prob = vec![1.0; joined_seq.len()];
165                let mut count = vec![1; joined_seq.len()];
166                for (i, kmer) in joined_seq.windows(kmer_size).enumerate() {
167                    let cn_kmer = skydive::utils::canonicalize_kmer(kmer);
168                    let score = **joined_scores.get(&cn_kmer).unwrap_or(&&1.0);
169
170                    for j in i..std::cmp::min(i + kmer_size, prob.len()) {
171                        prob[j] *= score;
172                        count[j] += 1;
173                    }
174                }
175
176                let mut qual = vec![0; joined_seq.len()];
177                for i in 0..prob.len() {
178                    prob[i] = prob[i].powf(1.0 / count[i] as f32);
179                    qual[i] = ((-10.0 * (1.0 - (prob[i] - 0.0001).max(0.0)).log10()) as u8 + 33)
180                        .clamp(33, 73);
181                }
182
183                let _ = writeln!(
184                    file,
185                    "@{}\n{}\n+\n{}",
186                    id,
187                    String::from_utf8(joined_seq).unwrap(),
188                    String::from_utf8_lossy(&qual)
189                );
190            }
191        });
192}

194fn pad_interval(start: &u64, end: &u64, window: usize) -> (u64, u64) {
195    // Calculate how much padding is needed to make the interval cleanly divisible by 1000
196    let interval_length = end - start;
197    let remainder = interval_length % window as u64;
198
199    let (padded_start, padded_end) = if remainder > 0 {
200        let padding_needed = window as u64 - remainder;
201        let half_padding = padding_needed / 2;
202
203        // Add padding evenly to start and end
204        let new_start = start.saturating_sub(half_padding);
205        let new_end = end + (padding_needed - half_padding);
206
207        (new_start, new_end)
208    } else {
209        (*start, *end)
210    };
211
212    (padded_start, padded_end)
