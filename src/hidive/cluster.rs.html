21pub fn start(
22    output: &PathBuf,
23    sample_name: &String,
24    from_loci_list: &Vec<String>,
25    to_loci_list: &Vec<String>,
26    reference_fasta_path: &PathBuf,
27    bam_path: &PathBuf,
28) {
29    // Get the system's temporary directory path
30    let cache_path = std::env::temp_dir();
31    skydive::elog!("Intermediate data will be stored at {:?}.", cache_path);
32
33    // Parse reference sequence file path.
34    let reference_seq_urls = skydive::parse::parse_file_names(&[reference_fasta_path.clone()]);
35    let reference_seq_url = reference_seq_urls.iter().next().unwrap();
36    let fasta = skydive::stage::open_fasta(&reference_seq_url).unwrap();
37
38    // Parse BAM file path.
39    let bam_urls = skydive::parse::parse_file_names(&[bam_path.clone()]);
40    let bam_url = bam_urls.iter().next().unwrap();
41
42    // Iterate over loc
43    let from_loci = skydive::parse::parse_loci(from_loci_list, 0)
44        .into_iter()
45        .collect::<Vec<_>>();
46    let to_loci = skydive::parse::parse_loci(to_loci_list, 0)
47        .into_iter()
48        .collect::<Vec<_>>();
49
50    // Open FASTA file for writing.
51    let mut buf_writer = BufWriter::new(File::create(output).unwrap());
52    let mut fasta_writer = fasta::Writer::new(&mut buf_writer);
53
54    for ((from_chr, from_start, from_stop, from_name), (to_chr, to_start, to_stop, to_name)) in
55        from_loci.iter().zip(to_loci.iter())
56    {
57        skydive::elog!(
58            "Found locus {} ({}:{}-{}) -> {} ({}:{}-{})...",
59            from_name,
60            from_chr,
61            from_start,
62            from_stop,
63            to_name,
64            to_chr,
65            to_start,
66            to_stop
67        );
68    }
69
70    for ((from_chr, from_start, from_stop, from_name), (to_chr, to_start, to_stop, to_name)) in
71        from_loci.iter().zip(to_loci.iter())
72    {
73        skydive::elog!(
74            "Processing locus {} ({}:{}-{}) -> {} ({}:{}-{})...",
75            from_name,
76            from_chr,
77            from_start,
78            from_stop,
79            to_name,
80            to_chr,
81            to_start,
82            to_stop
83        );
84
85        // Extract the reference sequence for the current locus.
86        let reference_seq = fasta
87            .fetch_seq_string(
88                to_chr.clone(),
89                usize::try_from(*to_start).unwrap(),
90                usize::try_from(*to_stop).unwrap(),
91            )
92            .unwrap();
93
94        // Initialize aligner
95        let mut aligner = Aligner::builder()
96            .map_hifi()
97            .with_cigar()
98            .with_seq(reference_seq.as_bytes())
99            .unwrap();
100
101        aligner.mapopt.flag |= minimap2::ffi::MM_F_EQX as i64;
102
103        // The BAM reader gets renewed for each locus, but it's fast to open.
104        let mut bam = skydive::stage::open_bam(&bam_url).unwrap();
105
106        let mut read_vectors = Vec::new();
107        let mut reads = Vec::new();
108
109        let _ = bam.fetch(FetchDefinition::RegionString(
110            from_chr.as_bytes(),
111            *from_start as i64,
112            *from_stop as i64,
113        ));
114        for read in bam
115            .records()
116            .filter(|r| r.is_ok())
117            .flatten()
118            .filter(|r| !r.is_secondary() && !r.is_supplementary())
119        {
120            let mappings = aligner
121                .map(
122                    &read.seq().as_bytes(),
123                    false,
124                    false,
125                    None,
126                    None,
127                    Some(read.qname()),
128                )
129                .unwrap();
130
131            for mapping in mappings {
132                // skydive::elog!("{:?}", mapping);
133
134                let mut variants = BTreeMap::new();
135                let mut ref_pos = mapping.target_start as i64 + 1;
136
137                let cigar = mapping.alignment.unwrap().cigar_str.unwrap();
138                let mut trim_left = 0;
139                let mut trim_right = 0;
140
141                let re = regex::Regex::new(r"(\d+)([MIDSNX=])").unwrap();
142                for (cap_idx, cap) in re.captures_iter(&cigar).enumerate() {
143                    let cigar_len = cap[1].parse::<u32>().unwrap();
144                    match &cap[2] {
145                        "M" | "=" => {
146                            ref_pos += cigar_len as i64;
147                        }
148                        "I" => {
149                            if cigar_len > 5 {
150                                variants.insert(ref_pos, 1);
151                            }
152                        }
153                        "D" => {
154                            if cigar_len > 5 {
155                                variants.insert(ref_pos, 1);
156                            }
157                            ref_pos += cigar_len as i64;
158                        }
159                        "X" => {
160                            variants.insert(ref_pos, 1);
161                            ref_pos += cigar_len as i64;
162                        }
163                        "S" => {
164                            if cap_idx == 0 {
165                                trim_left = cigar_len as usize;
166                            } else {
167                                trim_right = cigar_len as usize;
168                            }
169                        }
170                        _ => {}
171                    }
172                }
173
174                let seq = if mapping.strand == minimap2::Strand::Forward {
175                    read.seq().as_bytes()
176                } else {
177                    read.seq().as_bytes().reverse_complement()
178                };
179                let subseq = seq[trim_left..seq.len() - trim_right].to_vec();
180
181                read_vectors.push(variants);
182                reads.push(subseq);
183            }
184        }
185
186        let all_positions = read_vectors
187            .iter()
188            .flat_map(|r| r.keys())
189            .collect::<HashSet<_>>()
190            .into_iter()
191            .sorted()
192            .cloned()
193            .collect::<Vec<_>>();
194
195        let pos_to_idx = all_positions
196            .iter()
197            .enumerate()
198            .map(|(i, p)| (p, i))
199            .collect::<BTreeMap<_, _>>();
200
201        let mut data = Vec::with_capacity(read_vectors.len());
202        for read in read_vectors.iter() {
203            let mut row = vec![0.0; all_positions.len()];
204            for (&pos, &value) in read {
205                if let Some(&idx) = pos_to_idx.get(&pos) {
206                    row[idx] = value as f32;
207                }
208            }
209            data.push(row);
210        }
211
212        // Configure HDBSCAN parameters
213        let params = HdbscanHyperParams::builder()
214            .min_cluster_size(2)
215            .allow_single_cluster(true)
216            .dist_metric(DistanceMetric::Euclidean)
217            .build();
218
219        // Create and run clusterer
220        let clusterer = Hdbscan::new(&data, params);
221        let cluster = clusterer.cluster();
222
223        if let Ok(labels) = cluster {
224            let unique_labels = labels.iter().unique().collect::<Vec<_>>();
225
226            for unique_label in unique_labels {
227                let mut sg = spoa::Graph::new();
228                let mut la = spoa::AlignmentEngine::new(AlignmentType::kOV, 5, -4, -8, -6, -8, -4);
229
230                for (label, read) in labels.iter().zip(reads.iter()) {
231                    if label == unique_label {
232                        let subseq = read.clone();
233
234                        let seq_cstr = std::ffi::CString::new(subseq.clone()).unwrap();
235                        let seq_qual = std::ffi::CString::new(vec![b'I'; subseq.len()]).unwrap();
236                        let a = la.align(seq_cstr.as_ref(), &sg);
237                        sg.add_alignment(&a, seq_cstr.as_ref(), seq_qual.as_ref());
238                    }
239                }
240
241                if *unique_label >= 0 {
242                    let consensus_cstr = sg.consensus();
243                    let name = format!("{}.{}.{}", from_name, sample_name, unique_label);
244
245                    let record = fasta::Record::with_attrs(
246                        name.as_str(),
247                        None,
248                        consensus_cstr.to_str().unwrap().as_bytes(),
249                    );
250                    fasta_writer.write_record(&record).unwrap();
251                }
252            }
253        }
254    }
255}
