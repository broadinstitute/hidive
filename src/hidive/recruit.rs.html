18// Import the skydive module, which contains the necessary functions for staging data
19
20pub fn start(
21    output: &PathBuf,
22    kmer_size: usize,
23    min_kmers_pct: usize,
24    fasta_paths: &Vec<PathBuf>,
25    seq_paths: &Vec<PathBuf>,
26) {
27    let fasta_urls = skydive::parse::parse_file_names(fasta_paths);
28    let seq_urls = skydive::parse::parse_file_names(seq_paths);
29
30    // Get the system's temporary directory path
31    let cache_path = std::env::temp_dir();
32    skydive::elog!("Intermediate data will be stored at {:?}.", cache_path);
33
34    // Read the FASTA files and prepare a hashmap of k-mers to search for in the other files.
35    let mut kmer_set = HashSet::new();
36    for fasta_url in fasta_urls {
37        let reader = Reader::from_file(fasta_url.to_file_path().unwrap()).unwrap();
38
39        for record in reader.records().flatten() {
40            let fw_seq = record.seq();
41            let rl_seq = skydive::utils::homopolymer_compressed(fw_seq);
42
43            let kmers = rl_seq
44                .windows(kmer_size)
45                .map(skydive::utils::canonicalize_kmer)
46                .collect::<HashSet<Vec<u8>>>();
47
48            kmer_set.extend(kmers);
49        }
50    }
51
52    // Read the FASTA files and search for the k-mers in each read.
53    let mut all_records = Vec::new();
54    for seq_url in seq_urls {
55        let reader = Reader::from_file(seq_url.to_file_path().unwrap()).unwrap();
56
57        let progress_bar =
58            skydive::utils::default_unbounded_progress_bar("Searching for similar reads (0 found)");
59
60        // Create some thread-safe counters.
61        let found_items = Arc::new(AtomicUsize::new(0));
62        let processed_items = Arc::new(AtomicUsize::new(0));
63        let progress_bar = Arc::new(progress_bar);
64
65        const UPDATE_FREQUENCY: usize = 1_000_000;
66
67        let records: Vec<_> = reader
68            .records()
69            .flatten()
70            .par_bridge()
71            .filter_map(|read| {
72                // Increment progress counters and update the progress bar every once in a while.
73                let current_processed = processed_items.fetch_add(1, Ordering::Relaxed);
74                if current_processed % UPDATE_FREQUENCY == 0 {
75                    progress_bar.set_message(format!(
76                        "Searching for similar reads ({} found)",
77                        found_items
78                            .load(Ordering::Relaxed)
79                            .to_formatted_string(&Locale::en),
80                    ));
81                    progress_bar.inc(UPDATE_FREQUENCY as u64);
82                }
83
84                // Count the number of k-mers found in our hashset from the long reads.
85                let fw_seq = read.seq();
86                let rl_seq = skydive::utils::homopolymer_compressed(fw_seq);
87
88                let num_kmers = rl_seq
89                    .par_windows(kmer_size)
90                    .map(|kmer| kmer_set.contains(&skydive::utils::canonicalize_kmer(kmer)))
91                    .filter(|&contains| contains)
92                    .count();
93
94                // If the number of k-mers found is greater than the threshold, add the read to the list of similar reads.
95                if num_kmers as f32 / rl_seq.len() as f32 > min_kmers_pct as f32 / 100.0 {
96                    let current_found = found_items.fetch_add(1, Ordering::Relaxed);
97                    if current_found % UPDATE_FREQUENCY == 0 {
98                        progress_bar.set_message(format!(
99                            "Searching for similar reads ({} found)",
100                            (current_found + 1).to_formatted_string(&Locale::en)
101                        ));
102                    }
103                    Some(read)
104                } else {
105                    None
106                }
107            })
108            .collect();
109
110        // Final update to ensure the last counts are displayed
111        let final_found = found_items.load(Ordering::Relaxed);
112        let final_processed = processed_items.load(Ordering::Relaxed);
113        progress_bar.set_message(format!(
114            "Found {} reads in {} reads total.",
115            final_found.to_formatted_string(&Locale::en),
116            final_processed.to_formatted_string(&Locale::en)
117        ));
118        progress_bar.finish();
119
120        // Add the reads to the list of all records.
121        all_records.extend(records);
122    }
123
124    let file = File::create(output).expect("Could not open file for writing.");
125    let mut writer: Box<dyn Write> = if output.to_str().unwrap().ends_with(".gz") {
126        Box::new(GzEncoder::new(file, Compression::default()))
127    } else {
128        Box::new(file)
129    };
130
131    for (i, record) in all_records.iter().enumerate() {
132        writeln!(writer, ">read_{}", i).expect("Could not write to file");
133        writeln!(
134            writer,
135            "{}",
136            String::from_utf8_lossy(record.seq())
137        )
138        .expect("Could not write to file");
139    }
140
141    skydive::elog!(
142        "Wrote {} reads to {}.",
143        all_records.len().to_formatted_string(&Locale::en),
144        output.to_str().unwrap()
145    );
146}
