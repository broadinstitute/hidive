21pub fn start(
22    output: &PathBuf,
23    kmer_size: usize,
24    iterations: usize,
25    // test_split: f32,
26    long_read_seq_paths: &Vec<PathBuf>,
27    short_read_seq_paths: &Vec<PathBuf>,
28    truth_seq_paths: &Vec<PathBuf>,

30    test_short_read_seq_paths: &Vec<PathBuf>,
31    test_truth_seq_paths: &Vec<PathBuf>,
32    debug: bool,
33) {
34    let long_read_seq_urls = skydive::parse::parse_file_names(long_read_seq_paths);
35    let short_read_seq_urls = skydive::parse::parse_file_names(short_read_seq_paths);

54    //     .collapse()
55    //     .assemble_all();
56
57    let lr_contigs = l1.assemble_all();
58    let lr_distances = distance_to_a_contig_end(&lr_contigs, kmer_size);
59
60    let sr_contigs = s1.assemble_all();
61    let sr_distances = distance_to_a_contig_end(&sr_contigs, kmer_size);
62
63    // Read all test long reads.
64    let all_test_lr_seqs: Vec<Vec<u8>> = process_reads(&test_long_read_seq_urls, "test long");

66
67    // Read all test short reads.
68    let all_test_sr_seqs: Vec<Vec<u8>> = process_reads(&test_short_read_seq_urls, "test short");
69    let test_s1 = LdBG::from_sequences(String::from("test_s1"), kmer_size, &all_test_sr_seqs);
70
71    // Read all test truth sequences.
72    let all_test_truth_seqs: Vec<Vec<u8>> = process_reads(&test_truth_seq_urls, "test truth");
73    let test_t1 = LdBG::from_sequences(String::from("test_t1"), kmer_size, &all_test_truth_seqs);

75    let test_lr_contigs = test_l1.assemble_all();

77
78    let test_sr_contigs = test_s1.assemble_all();

80
81    // Configure GBDT.

84    cfg.set_max_depth(4);
85    cfg.set_min_leaf_size(5);
86    cfg.set_loss(&loss2string(&Loss::BinaryLogitraw));
87    cfg.set_shrinkage(0.05);
88    cfg.set_iterations(iterations);
89    cfg.set_debug(debug);

91    skydive::elog!("Training GBDT model with:");
92    skydive::elog!(" - feature_size={}", cfg.feature_size);
93    skydive::elog!(" - max_depth={}", cfg.max_depth);
94    skydive::elog!(" - min_leaf_size={}", cfg.min_leaf_size);
95    skydive::elog!(" - loss={}", loss2string(&cfg.loss));
96    skydive::elog!(" - shrinkage={}", cfg.shrinkage);
97    skydive::elog!(" - iterations={}", cfg.iterations);
98    // skydive::elog!(" - train/test={}/{}", 1.0 - test_split, test_split); //**** Commented out
99
100    // Initialize GBDT algorithm.
101    let mut gbdt = GBDT::new(&cfg);
102
103    // let mut training_data: DataVec = Vec::new();
104    // let mut test_data: DataVec = Vec::new();
105
106    // let mut rng = rand::rngs::StdRng::seed_from_u64(0);
107
108    let kmers = l1
109        .kmers
110        .keys()
111        .chain(s1.kmers.keys())
112        .chain(t1.kmers.keys());
113
114    let mut training_data: DataVec =
115        create_dataset_for_model(kmers, &lr_distances, &sr_distances, &l1, &s1, &t1);
116
117    // Train the decision trees.
118    skydive::elog!(
119        "Training GBDT model with {} training points...",
120        training_data.len().to_formatted_string(&Locale::en)
121    );
122    gbdt.fit(&mut training_data);
123
124    // Prepare test data.
125    let test_kmers = test_l1
126        .kmers
127        .keys()
128        .chain(test_s1.kmers.keys())
129        .chain(test_t1.kmers.keys());
130
131    let test_data: DataVec = create_dataset_for_model(
132        test_kmers,
133        &test_lr_distances,
134        &test_sr_distances,
135        &test_l1,
136        &test_s1,
137        &test_t1,
138    );
139
140    // Predict the test data.
141    skydive::elog!("Computing accuracy on test data...");
142    let prediction = gbdt.predict(&test_data);
143    let pred_threshold: f32 = 0.5;
144
145    // Evaluate accuracy of the model on the test data.
146    let mut num_correct = 0u32;
147    let mut num_total = 0u32;
148    for (data, pred) in test_data.iter().zip(prediction.iter()) {
149        let truth = data.label;
150        let call = if *pred > pred_threshold { 1.0 } else { 0.0 };
151        if (call - truth).abs() < f32::EPSILON {
152            num_correct += 1;
153        }
154        num_total += 1;
155    }
156
157    // Precision, Recall, and F1 score calculations.
158    let (precision, recall, f1_score) =
159        compute_precision_recall_f1(&test_data, &prediction, pred_threshold);
160    skydive::elog!("Prediction threshold: {:.2}", pred_threshold);
161    skydive::elog!("Precision: {:.2}%", 100.0 * precision);
162    skydive::elog!("Recall: {:.2}%", 100.0 * recall);
163    skydive::elog!("F1 score: {:.2}%", 100.0 * f1_score);
164
165    // TPR and FPR calculations at various thresholds.
166    skydive::elog!("Computing TPR and FPR at various thresholds...");
167    let fpr_tpr: Vec<(f32, f32)> = compute_fpr_tpr(&test_data, &prediction);
168
169    // Save TPR and FPR at various thresholds to a file.
170    let csv_output = output.with_extension("csv");
171    let mut writer = std::fs::File::create(csv_output).expect("Unable to create file");
172    for (tpr, fpr) in &fpr_tpr {
173        writeln!(writer, "{},{}", tpr, fpr).expect("Unable to write data");
174    }
175    skydive::elog!(
176        "TPR and FPR at various thresholds saved to {}",
177        output.to_str().unwrap()
178    );
179
180    // Create a ROC curve.
181    let png_output = output.with_extension("png");
182    plot_roc_curve(&png_output, &fpr_tpr).expect("Unable to plot ROC curve");
183    skydive::elog!("ROC curve saved to {}", png_output.to_str().unwrap());
184
185    skydive::elog!(
186        "Predicted accuracy: {}/{} ({:.2}%)",
187        num_correct.to_formatted_string(&Locale::en),
188        num_total.to_formatted_string(&Locale::en),
189        100.0 * num_correct as f32 / num_total as f32
190    );
191
192    gbdt.save_model(output.to_str().unwrap())
193        .expect("Unable to save model");
194
195    skydive::elog!("Model saved to {}", output.to_str().unwrap());
196}

198pub fn distance_to_a_contig_end(
199    contigs: &Vec<Vec<u8>>,
200    kmer_size: usize,
201) -> HashMap<Vec<u8>, usize> {
202    let mut distances = HashMap::new();
203
204    for contig in contigs {
205        for (distance_from_start, cn_kmer) in
206            contig.windows(kmer_size).map(canonicalize_kmer).enumerate()
207        {
208            let distance_from_end = contig.len() - distance_from_start - kmer_size;
209
210            distances.insert(
211                cn_kmer,
212                if distance_from_start < distance_from_end {
213                    distance_from_start
214                } else {
215                    distance_from_end
216                },
217            );
218        }
219    }
220
221    distances
222}

224pub fn process_reads(read_seq_urls: &HashSet<Url>, read_type: &str) -> Vec<Vec<u8>> {
225    let mut all_seqs: Vec<Vec<u8>> = Vec::new();
226    for read_seq_url in read_seq_urls {
227        let basename = skydive::utils::basename_without_extension(
228            read_seq_url,
229            &[".fasta.gz", ".fa.gz", ".fasta", ".fa"],
230        );
231        let fasta_path = read_seq_url.to_file_path().unwrap();
232
233        skydive::elog!("Processing {}-read sample {}...", read_type, basename);
234
235        let reader = bio::io::fasta::Reader::from_file(&fasta_path).unwrap();
236        let all_reads: Vec<bio::io::fasta::Record> = reader.records().map(|r| r.unwrap()).collect();
237
238        all_seqs.extend(
239            all_reads
240                .iter()
241                .map(|r| r.seq().to_vec())
242                .collect::<Vec<Vec<u8>>>(),
243        );
244    }
245
246    all_seqs
247}

249pub fn plot_roc_curve(
250    output: &PathBuf,
251    roc_points: &[(f32, f32)],
252) -> Result<(), Box<dyn std::error::Error>> {

254    root.fill(&WHITE)?;
255
256    let mut chart = ChartBuilder::on(&root)
257        .caption("ROC Curve", ("sans-serif", 50).into_font())
258        .margin(10)
259        .x_label_area_size(30)
260        .y_label_area_size(50)
261        .build_cartesian_2d(0.0..1.0, 0.0..1.0)?;
262
263    chart
264        .configure_mesh()
265        .x_desc("False Positive Rate")
266        .y_desc("True Positive Rate")
267        .draw()?;
268
269    chart
270        .draw_series(LineSeries::new(
271            roc_points
272                .iter()
273                .map(|(fpr, tpr)| (*fpr as f64, *tpr as f64)),
274            &RED,
275        ))?
276        .label("ROC Curve")
277        .legend(|(x, y)| PathElement::new([(x, y), (x + 20, y)], RED));
278
279    // Draw a diagonal dotted line for reference.
280    chart
281        .draw_series(LineSeries::new(
282            [(0.0, 0.0), (1.0, 1.0)].iter().cloned(),
283            &BLACK,
284        ))?
285        .label("Random")
286        .legend(|(x, y)| PathElement::new(vec![(x, y), (x + 20, y)], BLACK));
287
288    chart
289        .configure_series_labels()
290        .background_style(WHITE)
291        .draw()?;
292
293    Ok(())
294}

296/// Computes FPR and TPR at various thresholds.
297pub fn compute_fpr_tpr(test_data: &DataVec, p: &Vec<f32>) -> Vec<(f32, f32)> {
298    skydive::elog!("Computing FPR and TPR at various thresholds...");
299    let mut fpr_tpr: Vec<(f32, f32)> = Vec::new();
300
301    // Iterate over different thresholds
302    for threshold in 0..100 {
303        let pred_threshold = threshold as f32 / 100.0;
304        let (
305            mut num_true_positives,
306            mut num_false_positives,
307            mut num_false_negatives,
308            mut num_true_negatives,
309        ) = (0u32, 0u32, 0u32, 0u32);
310
311        // Iterate over test data and predictions
312        for (data, pred) in test_data.iter().zip(p.iter()) {
313            let truth = data.label;
314            let call = if *pred > pred_threshold { 1.0 } else { 0.0 };
315
316            if (call - truth).abs() < f32::EPSILON {
317                // Correct classification
318                if truth > 0.5 {
319                    num_true_positives += 1;
320                }
321                // True Positive
322                else {
323                    num_true_negatives += 1;
324                } // True Negative
325            } else {
326                // Incorrect classification
327                if truth < 0.5 {
328                    num_false_positives += 1;
329                } // False Positive
330                if truth > 0.5 {
331                    num_false_negatives += 1;
332                } // False Negative
333            }
334        }
335
336        // Calculate FPR (False Positive Rate)
337        let fpr = if num_false_positives + num_true_negatives > 0 {
338            num_false_positives as f32 / (num_false_positives + num_true_negatives) as f32
339        } else {
340            0.0
341        };
342
343        // Calculate TPR (True Positive Rate)
344        let tpr = if num_true_positives + num_false_negatives > 0 {
345            num_true_positives as f32 / (num_true_positives + num_false_negatives) as f32
346        } else {
347            0.0
348        };
349
350        fpr_tpr.push((fpr, tpr));
351    }
352
353    fpr_tpr
354}

356/// Computes precision, recall, and F1 score on test data.
357pub fn compute_precision_recall_f1(
358    test_data: &DataVec,
359    p: &Vec<f32>,
360    pred_threshold: f32,
361) -> (f32, f32, f32) {
362    skydive::elog!("Computing precision, recall, and F1 score on test data...");
363    let (mut num_true_positives, mut num_false_positives, mut num_false_negatives) =
364        (0u32, 0u32, 0u32);
365
366    for (data, pred) in test_data.iter().zip(p.iter()) {
367        let truth = data.label;
368        let call = if *pred > pred_threshold { 1.0 } else { 0.0 };
369        if (call - truth).abs() < f32::EPSILON {
370            if truth > 0.5 {
371                num_true_positives += 1;
372            }
373        } else {
374            if truth < 0.5 {
375                num_false_positives += 1;
376            }
377            if truth > 0.5 {
378                num_false_negatives += 1;
379            }
380        }
381    }
382
383    let precision = num_true_positives as f32 / (num_true_positives + num_false_positives) as f32;
384    let recall = num_true_positives as f32 / (num_true_positives + num_false_negatives) as f32;
385    let f1_score = 2.0 * precision * recall / (precision + recall);
386
387    (precision, recall, f1_score)
388}

390/// Creates a dataset for the ML model.
391pub fn create_dataset_for_model(
392    kmers: Chain<Chain<Keys<Vec<u8>, Record>, Keys<Vec<u8>, Record>>, Keys<Vec<u8>, Record>>,
393    lr_distances: &HashMap<Vec<u8>, usize>,
394    sr_distances: &HashMap<Vec<u8>, usize>,
395    l1: &LdBG,
396    s1: &LdBG,
397    t1: &LdBG,
398) -> Vec<Data> {
399    let mut dataset = Vec::new();
400
401    for kmer in kmers {
402        let compressed_len = skydive::utils::homopolymer_compressed(kmer).len();
403        let compressed_len_diff = (kmer.len() - compressed_len) as f32;
404        let entropy = skydive::utils::shannon_entropy(kmer);
405        let gc_content = skydive::utils::gc_content(kmer);
406        // let lr_distance = *lr_distances.get(kmer).unwrap_or(&0) as f32;
407        // let sr_distance = *sr_distances.get(kmer).unwrap_or(&0) as f32;
408
409        let lcov = l1.kmers.get(kmer).map_or(0, |lr| lr.coverage());
410
411        let scov_fw = s1.kmers.get(kmer).map_or(0, |sr| sr.fw_coverage());
412        let scov_rc = s1.kmers.get(kmer).map_or(0, |sr| sr.rc_coverage());
413        let scov_total = scov_fw + scov_rc;
414        let strand_ratio = if scov_total > 0 {
415            (scov_fw as f32).max(scov_rc as f32) / scov_total as f32
416        } else {
417            0.5
418        };
419
420        // let expected = (scov_fw + scov_rc) as f32 / 2.0;
421        // let chi_square = if expected > 0.0 {
422        //     ((scov_fw as f32 - expected).powi(2) +
423        //     (scov_rc as f32 - expected).powi(2)) / expected
424        // } else {
425        //     0.0
426        // };
427
428        let tcov = t1.kmers.get(kmer).map_or(0, |tr| tr.coverage());
429
430        let data = Data::new_training_data(
431            vec![
432                if lcov > 0 { 1.0 } else { 0.0 }, // present in long reads
433                scov_total as f32,                // coverage in short reads
434                strand_ratio as f32, // measure of strand bias (0.5 = balanced, 1.0 = all on one strand)
435                compressed_len_diff, // homopolymer compression length difference
436                entropy,             // shannon entropy
437                gc_content,          // gc content
438                                     // lr_distance,                         // distance to nearest long read contig end
439                                     // sr_distance,                         // distance to nearest short read contig end
440            ],
441            1.0,
442            if tcov > 0 { 1.0 } else { 0.0 }, // present in truth
443            None,
444        );
445
446        dataset.push(data);
447    }
448
449    dataset
